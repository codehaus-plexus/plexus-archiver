154         throws Exception
155     {
156 
157         String[] executablePaths =
158         {
159             "path/to/executable", "path/to/executable.bat"
160         };
161 
162         String[] confPaths =
163         {
164             "path/to/etc/file", "path/to/etc/file2"
165         };
166 
167         String[] logPaths =
168         {
169             "path/to/logs/log.txt"
170         };
171 
172         int exeMode = 0777;
173         int confMode = 0600;
174         int logMode = 0640;
175 
176         File tmpDir = null;
177         try
178         {
179             tmpDir = File.createTempFile( "zip-with-chmod.", ".dir" );
180             tmpDir.delete();
181 
182             tmpDir.mkdirs();
183 
184             for ( String executablePath : executablePaths )
185             {
186                 writeFile( tmpDir, executablePath, exeMode );
187             }
188 
189             for ( String confPath : confPaths )
190             {
191                 writeFile( tmpDir, confPath, confMode );
192             }
193 
194             for ( String logPath : logPaths )
195             {
196                 writeFile( tmpDir, logPath, logMode );
197             }
198 
199             {
200                 Map<String, PlexusIoResourceAttributes> attributesByPath =
201                     PlexusIoResourceAttributeUtils.getFileAttributesByPath( tmpDir );
202                 for ( String path : executablePaths )
203                 {
204                     PlexusIoResourceAttributes attrs = attributesByPath.get( path );
205                     if ( attrs == null )
206                     {
207                         attrs = attributesByPath.get( new File( tmpDir, path ).getAbsolutePath() );
208                     }
209 
210                     assertNotNull( attrs );
211                     assertEquals( exeMode, attrs.getOctalMode(), "Wrong mode for: " + path );
212                 }
213 
214                 for ( String path : confPaths )
215                 {
216                     PlexusIoResourceAttributes attrs = attributesByPath.get( path );
217                     if ( attrs == null )
218                     {
219                         attrs = attributesByPath.get( new File( tmpDir, path ).getAbsolutePath() );
220                     }
221 
222                     assertNotNull( attrs );
223                     assertEquals( confMode, attrs.getOctalMode(), "Wrong mode for: " + path );
224                 }
225 
226                 for ( String path : logPaths )
227                 {
228                     PlexusIoResourceAttributes attrs = attributesByPath.get( path );
229                     if ( attrs == null )
230                     {
231                         attrs = attributesByPath.get( new File( tmpDir, path ).getAbsolutePath() );
232                     }
233 
234                     assertNotNull( attrs );
235                     assertEquals( logMode, attrs.getOctalMode(), "Wrong mode for: " + path );
236                 }
237             }
238 
239             File zipFile = getTestFile( "target/output/zip-with-modes.zip" );
240 
241             ZipArchiver archiver = getZipArchiver( zipFile );
242 
243             archiver.addDirectory( tmpDir );
244             archiver.createArchive();
245 
246             assertTrue( zipFile.exists() );
247 
248             File zipFile2 = getTestFile( "target/output/zip-with-modes-L2.zip" );
249 
250             archiver = getZipArchiver();
251             archiver.setDestFile( zipFile2 );
252 
253             archiver.addArchivedFileSet( zipFile );
254             archiver.createArchive();
255 
256             ZipFile zf = new ZipFile( zipFile2 );
257 
258             for ( String path : executablePaths )
259             {
260                 ZipArchiveEntry ze = zf.getEntry( path );
261 
262                 int mode = ze.getUnixMode() & UnixStat.PERM_MASK;
263 
264                 assertEquals( exeMode, mode, "Wrong mode for: " + path );
265             }
266 
267             for ( String path : confPaths )
268             {
269                 ZipArchiveEntry ze = zf.getEntry( path );
270 
271                 int mode = ze.getUnixMode() & UnixStat.PERM_MASK;
272 
273                 assertEquals( confMode, mode, "Wrong mode for: " + path );
274             }
275 
276             for ( String path : logPaths )
277             {
278                 ZipArchiveEntry ze = zf.getEntry( path );
279 
280                 int mode = ze.getUnixMode() & UnixStat.PERM_MASK;
281 
282                 assertEquals( logMode, mode, "Wrong mode for: " + path );
283             }
284         }
285         finally
286         {
287             if ( tmpDir != null && tmpDir.exists() )
288             {
289                 try
290                 {
291                     FileUtils.forceDelete( tmpDir );
292                 }
293                 catch ( IOException e )
294                 {
295                     e.printStackTrace();
296                 }
297             }
298         }

372     @Test
373     public void testRecompressAddedZips() throws Exception
374     {
375         // check that by default the zip archives are re-compressed
376 
377         final File zipFileRecompress = getTestFile( "target/output/recompress-added-zips.zip" );
378         final ZipArchiver zipArchiverRecompress = getZipArchiver( zipFileRecompress );
379         zipArchiverRecompress.addDirectory( getTestFile( "src/test/jars" ) );
380         FileUtils.removePath( zipFileRecompress.getPath() );
381         zipArchiverRecompress.createArchive();
382 
383         final ZipFile zfRecompress = new ZipFile( zipFileRecompress );
384         assertEquals( ZipEntry.DEFLATED, zfRecompress.getEntry( "test.zip" ).getMethod() );
385         assertEquals( ZipEntry.DEFLATED, zfRecompress.getEntry( "test.jar" ).getMethod() );
386         assertEquals( ZipEntry.DEFLATED, zfRecompress.getEntry( "test.rar" ).getMethod() );
387         assertEquals( ZipEntry.DEFLATED, zfRecompress.getEntry( "test.tar.gz" ).getMethod() );
388         zfRecompress.close();
389 
390         // make sure the zip files are not re-compressed when recompressAddedZips is set to false
391 
392         final File zipFileDontRecompress = getTestFile( "target/output/dont-recompress-added-zips.zip" );
393         ZipArchiver zipArchiver = getZipArchiver( zipFileDontRecompress );
394         zipArchiver.addDirectory( getTestFile( "src/test/jars" ) );
395         zipArchiver.setRecompressAddedZips( false );
396         FileUtils.removePath( zipFileDontRecompress.getPath() );
397         zipArchiver.createArchive();
398 
399         final ZipFile zfDontRecompress = new ZipFile( zipFileDontRecompress );
400         final ZipArchiveEntry zipEntry = zfDontRecompress.getEntry( "test.zip" );
401         final ZipArchiveEntry jarEntry = zfDontRecompress.getEntry( "test.jar" );
402         final ZipArchiveEntry rarEntry = zfDontRecompress.getEntry( "test.rar" );
403         final ZipArchiveEntry tarEntry = zfDontRecompress.getEntry( "test.tar.gz" );
404         // check if only zip files are not compressed...
405         assertEquals( ZipEntry.STORED, zipEntry.getMethod() );
406         assertEquals( ZipEntry.STORED, jarEntry.getMethod() );
407         assertEquals( ZipEntry.STORED, rarEntry.getMethod() );
408         assertEquals( ZipEntry.DEFLATED, tarEntry.getMethod() );
409         // ...and no file is corrupted in the process
410         assertTrue( IOUtil.contentEquals( Files.newInputStream( getTestFile( "src/test/jars/test.zip" ).toPath() ),
411                     zfDontRecompress.getInputStream( zipEntry ) ) );
412         assertTrue( IOUtil.contentEquals( Files.newInputStream( getTestFile( "src/test/jars/test.jar" ).toPath() ),
413                 zfDontRecompress.getInputStream( jarEntry ) ) );
414         assertTrue( IOUtil.contentEquals( Files.newInputStream( getTestFile( "src/test/jars/test.rar" ).toPath() ),
415                 zfDontRecompress.getInputStream( rarEntry ) ) );
416         assertTrue( IOUtil.contentEquals( Files.newInputStream( getTestFile( "src/test/jars/test.tar.gz" ).toPath() ),
417                 zfDontRecompress.getInputStream( tarEntry ) ) );
418         zfDontRecompress.close();
419     }

526     private void createArchive( ZipArchiver archiver )
527         throws ArchiverException, IOException
528     {
529         archiver.createArchive();
530 
531         ZipFile zf = new ZipFile( archiver.getDestFile() );
532 
533         Enumeration e = zf.getEntries();
534 
535         while ( e.hasMoreElements() )
536         {
537             ZipArchiveEntry ze = (ZipArchiveEntry) e.nextElement();
538             if ( ze.isDirectory() )
539             {
540                 if ( ze.getName().startsWith( "worldwritable" ) )
541                 {
542                     fileModeAssert( 0777, UnixStat.PERM_MASK & ze.getUnixMode() );
543                 }
544                 else if ( ze.getName().startsWith( "groupwritable" ) )
545                 {
546                     fileModeAssert( 0070, UnixStat.PERM_MASK & ze.getUnixMode() );
547                 }
548                 else
549                 {
550                     fileModeAssert( 0500, UnixStat.PERM_MASK & ze.getUnixMode() );
551                 }
552             }
553             else
554             {
555                 if ( ze.getName().equals( "one.txt" ) )
556                 {
557                     fileModeAssert( 0640, UnixStat.PERM_MASK & ze.getUnixMode() );
558                 }
559                 else if ( ze.getName().equals( "two.txt" ) )
560                 {
561                     fileModeAssert( 0664, UnixStat.PERM_MASK & ze.getUnixMode() );
562                 }
563                 else if ( ze.isUnixSymlink() )
564                 {
565                     //         assertEquals( ze.getName(), 0500, UnixStat.PERM_MASK & ze.getUnixMode() );
566                 }
567                 else
568                 {
569                     fileModeAssert( 0400, UnixStat.PERM_MASK & ze.getUnixMode() );
570                 }
571             }
572 
573         }
574     }

638      * Zip archives store file modification times with a granularity of two seconds.
639      * Verify that ZipArchiver rounds up the last modified time.
640      */
641     @Test
642     public void testLastModifiedTimeRounding()
643         throws Exception
644     {
645         Path oddSecondsTimestampFile = Files.createTempFile( "odd-seconds-timestamp", null );
646         oddSecondsTimestampFile.toFile().deleteOnExit();
647         // The milliseconds part is set to zero as not all filesystem support timestamp more granular than second.
648         Files.setLastModifiedTime( oddSecondsTimestampFile, FileTime.fromMillis( 1534189011_000L ) );
649         Path evenSecondsTimestampFile = Files.createTempFile( "even-seconds-timestamp", null );
650         evenSecondsTimestampFile.toFile().deleteOnExit();
651         Files.setLastModifiedTime( evenSecondsTimestampFile, FileTime.fromMillis( 1534189012_000L ) );
652 
653         File destFile = getTestFile( "target/output/last-modified-time.zip" );
654         ZipArchiver archiver = getZipArchiver( destFile );
655         archiver.addFile( oddSecondsTimestampFile.toFile(), "odd-seconds" );
656         archiver.addFile( evenSecondsTimestampFile.toFile(), "even-seconds" );
657         archiver.createArchive();
658 
659         // verify that the last modified time of the entry is equal or newer than the original file
660         try ( ZipFile resultingZipFile = new ZipFile( destFile ) )
661         {
662             assertEquals( 1534189012_000L, resultingZipFile.getEntry( "odd-seconds" ).getTime() );
663             assertEquals( 1534189012_000L, resultingZipFile.getEntry( "even-seconds" ).getTime() );
664 
665             FileTime expected = FileTime.fromMillis( 1534189012_000L );
666             assertEquals( expected, resultingZipFile.getEntry( "odd-seconds" ).getLastModifiedTime() );
667             assertEquals( expected, resultingZipFile.getEntry( "even-seconds" ).getLastModifiedTime() );
668         }

940     /**
941      * Takes a timestamp, turns it into a textual representation based on GMT, then translated it into a timestamp in
942      * local timezone. This makes the test independent of the current TimeZone. The reason this is necessary is:
943      * <ul>
944      * <li>ZIP file format does not take timezone into account.</li>
945      * <li>In the process of converting the ZipEntry time from the DOS date format specified by the ZIP file format, the
946      * timestamp is converted to a Java Date object, which DOES depends of the current system TimeZone, therefore
947      * changing the value of the Date object representing that timestamp relative to the local TimeZone.</li>
948      * </ul>
949      *
950      * @param timestamp the epoch time to convert.
951      * @return the timestamp matching the same input date but in the local TZ.
952      */
953     private long toLocalTimeZone( long timestamp )
954     {
955         String dateFormat = "dd-MM-yyyy hh:mm:ss a";
956         DateFormat formatterWithTimeZone = new SimpleDateFormat( dateFormat );
957         formatterWithTimeZone.setTimeZone( TimeZone.getTimeZone( "GMT" ) );
958         String sDate = formatterWithTimeZone.format( new Date( timestamp ) );
959 
960         DateFormat formatter = new SimpleDateFormat( dateFormat );
961         try
962         {
963             Date dateWithTimeZone = formatter.parse( sDate );
964             return dateWithTimeZone.getTime();
965         }
966         catch ( ParseException e )
967         {
968             fail( "Date '" + sDate + "' can not be parsed!" );
969             return 0L;
970         }
971     }
