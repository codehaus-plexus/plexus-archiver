26      /**
27       * Iterate over the TarFile and run the consumer on each TarArchiveEntry.
28       */
29      public static void forEachTarArchiveEntry( TarFile file, Consumer<TarArchiveEntry> consumer )
30      {
31          try
32          {
33              Collections.list( file.getEntries() )
34                  .stream()
35                  .filter( entry -> !entry.isDirectory() )
36                  .map( TarArchiveEntry.class::cast )
37                  .forEachOrdered( consumer );
38          }
39          catch ( IOException ioe )
40          {
41              throw new UncheckedIOException( ioe );
42          }
43      }
44  
45      private static void forEachZipArchiveEntry( ZipFile file, Consumer<ZipArchiveEntry> consumer )
46      {
47          Collections.list( file.getEntries() )
48              .stream()
49              .filter( entry -> !entry.isDirectory() )
50              .forEachOrdered( consumer );
51      }

53      /**
54       * Creates a map with the archive files contents. The map keys are the names of file entries in the archive file.
55       * The map values are the respective archive entries.
56       */
57      private static Map<String, TarArchiveEntry> getFileEntries( TarFile file )
58      {
59          try
60          {
61              return Collections.list( file.getEntries() )
62                  .stream()
63                  .filter( entry -> !entry.isDirectory() )
64                  .map( TarArchiveEntry.class::cast )
65                  .collect( Collectors.toMap( ArchiveEntry::getName, Function.identity() ) );
66          }
67          catch ( IOException e )
68          {
69              throw new UncheckedIOException( e );
70          }
71      }

73      /**
74       * Creates a map with the archive files contents. The map keys are the names of file entries in the archive file.
75       * The map values are the respective archive entries.
76       */
77      private static Map<String, ZipArchiveEntry> getFileEntries( ZipFile file )
78      {
79          return Collections.list( file.getEntries() )
80              .stream()
81              .filter( entry -> !entry.isDirectory() )
82              .collect( Collectors.toMap( ArchiveEntry::getName, Function.identity() ) );
83      }

85      private static void assertTarEquals( TarFile file1, TarArchiveEntry entry1,
86                                           TarFile file2, TarArchiveEntry entry2 )
87      {
88          assertThat( entry1.isDirectory() ).isEqualTo( entry2.isDirectory() );
89          assertThat( entry1.isSymbolicLink() ).isEqualTo( entry2.isSymbolicLink() );
90          assertThat( entry1.getLastModifiedDate() ).isEqualTo( entry2.getLastModifiedDate() );
91          assertThat( entry1.getUserName() ).isEqualTo( entry2.getUserName() );
92          assertThat( entry1.getGroupName() ).isEqualTo( entry2.getGroupName() );
93  
94          try ( InputStream is1 = file1.getInputStream( entry1 );
95                InputStream is2 = file2.getInputStream( entry2 ); )
96          {
97              assertThat( is1 )
98                  .as( "Content of the entry1 %s is different of entry2 %s", entry1.getName(), entry2.getName() )
99                  .hasSameContentAs( is2 );
100         }
101         catch ( IOException ioe )
102         {
103             throw new UncheckedIOException( ioe );
104         }
105     }

107     private static void assertTarZipEquals( TarFile file1, TarArchiveEntry entry1,
108                                             ZipFile file2, ZipArchiveEntry entry2 )
109     {
110         assertThat( entry1.isDirectory() ).isEqualTo( entry2.isDirectory() );
111         assertThat( entry1.getLastModifiedDate() ).isCloseTo( entry2.getLastModifiedDate(), 1000 );
112

146     /**
147      * Called to compare the given Tar files.
148      */
149     public static void assertTarEquals( final TarFile file1, final TarFile file2, final String prefix )
150     {
151         final Map<String, TarArchiveEntry> map2 = getFileEntries( file2 );
152 
153         forEachTarArchiveEntry( file1, entry1 -> {
154             final String name2 = getNameWithPrefix( prefix, entry1.getName() );
155             final TarArchiveEntry entry2 = map2.remove( name2 );
156             assertThat( entry2 ).as( "Missing entry in file2: %s", name2 ).isNotNull();
157             assertTarEquals( file1, entry1, file2, entry2 );
158         } );
159 
160         assertThat( map2 ).as("Found additional entries in file2").isEmpty();
161     }
