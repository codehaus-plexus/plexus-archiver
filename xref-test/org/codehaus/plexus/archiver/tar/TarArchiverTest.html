71      @Test
72      @DisabledOnOs( OS.WINDOWS )
73      public void testCreateArchiveWithDetectedModes()
74          throws Exception
75      {
76  
77          String[] executablePaths =
78          {
79              "path/to/executable", "path/to/executable.bat"
80          };
81  
82          String[] confPaths =
83          {
84              "path/to/etc/file", "path/to/etc/file2"
85          };
86  
87          String[] logPaths =
88          {
89              "path/to/logs/log.txt"
90          };
91  
92          int exeMode = 0777;
93          int confMode = 0600;
94          int logMode = 0640;
95  
96          File tmpDir = null;
97          try
98          {
99              tmpDir = File.createTempFile( "tbz2-with-chmod.", ".dir" );
100             tmpDir.delete();
101 
102             tmpDir.mkdirs();
103 
104             for ( String executablePath : executablePaths )
105             {
106                 writeFile( tmpDir, executablePath, exeMode );
107             }
108 
109             for ( String confPath : confPaths )
110             {
111                 writeFile( tmpDir, confPath, confMode );
112             }
113 
114             for ( String logPath : logPaths )
115             {
116                 writeFile( tmpDir, logPath, logMode );
117             }
118 
119             {
120                 Map attributesByPath = PlexusIoResourceAttributeUtils.getFileAttributesByPath( tmpDir );
121                 for ( String path : executablePaths )
122                 {
123                     PlexusIoResourceAttributes attrs = (PlexusIoResourceAttributes) attributesByPath.get( path );
124                     if ( attrs == null )
125                     {
126                         attrs = (PlexusIoResourceAttributes) attributesByPath.get(
127                             new File( tmpDir, path ).getAbsolutePath() );
128                     }
129 
130                     assertNotNull( attrs );
131                     assertEquals( exeMode, attrs.getOctalMode(), "Wrong mode for: " + path );
132                 }
133 
134                 for ( String path : confPaths )
135                 {
136                     PlexusIoResourceAttributes attrs = (PlexusIoResourceAttributes) attributesByPath.get( path );
137                     if ( attrs == null )
138                     {
139                         attrs = (PlexusIoResourceAttributes) attributesByPath.get(
140                             new File( tmpDir, path ).getAbsolutePath() );
141                     }
142 
143                     assertNotNull( attrs );
144                     assertEquals( confMode, attrs.getOctalMode(), "Wrong mode for: " + path );
145                 }
146 
147                 for ( String path : logPaths )
148                 {
149                     PlexusIoResourceAttributes attrs = (PlexusIoResourceAttributes) attributesByPath.get( path );
150                     if ( attrs == null )
151                     {
152                         attrs = (PlexusIoResourceAttributes) attributesByPath.get(
153                             new File( tmpDir, path ).getAbsolutePath() );
154                     }
155 
156                     assertNotNull( attrs );
157                     assertEquals( logMode, attrs.getOctalMode(), "Wrong mode for: " + path );
158                 }
159             }
160 
161             File tarFile = getTestFile( "target/output/tar-with-modes.tar" );
162 
163             TarArchiver archiver = getPosixTarArchiver();
164             archiver.setDestFile( tarFile );
165 
166             archiver.addDirectory( tmpDir );
167             archiver.createArchive();
168 
169             assertTrue( tarFile.exists() );
170 
171             File tarFile2 = getTestFile( "target/output/tar-with-modes-L2.tar" );
172 
173             archiver = getPosixTarArchiver();
174             archiver.setDestFile( tarFile2 );
175 
176             archiver.addArchivedFileSet( tarFile );
177             archiver.createArchive();
178 
179             TarFile tf = new TarFile( tarFile2 );
180 
181             Map<String, TarArchiveEntry> entriesByPath = new LinkedHashMap<String, TarArchiveEntry>();
182             for ( Enumeration e = tf.getEntries(); e.hasMoreElements(); )
183             {
184                 TarArchiveEntry te = (TarArchiveEntry) e.nextElement();
185                 entriesByPath.put( te.getName(), te );
186             }
187 
188             for ( String path : executablePaths )
189             {
190                 TarArchiveEntry te = entriesByPath.get( path );
191 
192                 int mode = te.getMode() & UnixStat.PERM_MASK;
193 
194                 assertEquals( exeMode, mode, "Wrong mode for: " + path );
195             }
196 
197             for ( String path : confPaths )
198             {
199                 TarArchiveEntry te = entriesByPath.get( path );
200 
201                 int mode = te.getMode() & UnixStat.PERM_MASK;
202 
203                 assertEquals( confMode, mode, "Wrong mode for: " + path );
204             }
205 
206             for ( String path : logPaths )
207             {
208                 TarArchiveEntry te = entriesByPath.get( path );
209 
210                 int mode = te.getMode() & UnixStat.PERM_MASK;
211 
212                 assertEquals( logMode, mode, "Wrong mode for: " + path );
213             }
214         }
215         finally
216         {
217             if ( tmpDir != null && tmpDir.exists() )
218             {
219                 try
220                 {
221                     FileUtils.forceDelete( tmpDir );
222                 }
223                 catch ( IOException e )
224                 {
225                     e.printStackTrace();
226                 }
227             }
228         }
229     }

261     private void writeFile( File dir, String fname, int mode )
262         throws IOException, ArchiverException
263     {
264         File file = new File( dir, fname );
265 
266         if ( file.getParentFile() != null )
267         {
268             file.getParentFile().mkdirs();
269         }
270 
271         try ( Writer writer = Files.newBufferedWriter( file.toPath(), StandardCharsets.UTF_8 ) )
272         {
273             writer.write( "This is a test file." );
274         }
275 
276         ArchiveEntryUtils.chmod( file, mode );
277     }

295     public void createArchive( final int directoryMode, final int fileModes[] )
296         throws Exception
297     {
298         int defaultFileMode = fileModes[0];
299         int oneFileMode = fileModes[1];
300         int twoFileMode = fileModes[2];
301 
302         TarArchiver archiver = getPosixTarArchiver();
303 
304         archiver.setDirectoryMode( directoryMode );
305 
306         archiver.setFileMode( defaultFileMode );
307 
308         archiver.addDirectory( getTestFile( "src/main" ) );
309         archiver.setFileMode( oneFileMode );
310 
311         archiver.addFile( getTestFile( "src/test/resources/manifests/manifest1.mf" ), "one.txt" );
312         archiver.addFile( getTestFile( "src/test/resources/manifests/manifest2.mf" ), "two.txt", twoFileMode );
313         archiver.setDestFile( getTestFile( "target/output/archive.tar" ) );
314 
315         archiver.addSymlink( "link_to_test_destinaton", "../test_destination/" );
316 
317         archiver.createArchive();
318 
319         TarArchiveInputStream tis;
320 
321         tis = new TarArchiveInputStream( bufferedInputStream( Files.newInputStream( archiver.getDestFile().toPath() ) ) );
322         TarArchiveEntry te;
323 
324         while ( ( te = tis.getNextTarEntry() ) != null )
325         {
326             if ( te.isDirectory() )
327             {
328                 assertEquals( directoryMode, te.getMode() & UnixStat.PERM_MASK,
329                               "un-expected tar-entry mode for [te.name=" + te.getName() + "]" );
330             }
331             else if ( te.isSymbolicLink() )
332             {
333                 assertEquals( "../test_destination/", te.getLinkName() );
334                 assertEquals( "link_to_test_destinaton", te.getName() );
335                 assertEquals( 0640, te.getMode() & UnixStat.PERM_MASK );
336             }
337             else
338             {
339                 if ( te.getName().equals( "one.txt" ) )
340                 {
341                     assertEquals( oneFileMode, te.getMode() & UnixStat.PERM_MASK );
342                 }
343                 else if ( te.getName().equals( "two.txt" ) )
344                 {
345                     assertEquals( twoFileMode, te.getMode() & UnixStat.PERM_MASK );
346                 }
347                 else
348                 {
349                     assertEquals( defaultFileMode, te.getMode() & UnixStat.PERM_MASK,
350                                   "un-expected tar-entry mode for [te.name=" + te.getName() + "]" );
351                 }
352 
353             }
354         }
355         IOUtil.close( tis );
356 
357     }

414         File createTarFile()
415             throws Exception
416         {
417             final File srcDir = new File( "src" );

446     private class GZipTarHandler
447         extends TarHandler
448     {
449 
450         @Override
451         File createTarFile()
452             throws Exception
453         {
454             File file = super.createTarFile();
455             File compressedFile = new File( file.getPath() + ".gz" );
456             Compressor compressor = new GZipCompressor();
457             compressor.setSource( createResource( file, file.getName() ) );
458             compressor.setDestFile( compressedFile );
459             compressor.compress();
460             compressor.close();
461             return compressedFile;
462         }
463 
464         @Override
465         TarFile newTarFile( File tarFile )
466         {
467             return new GZipTarFile( tarFile );
468         }
