278     public void createArchive(final int directoryMode, final int fileModes[])
279         throws Exception
280     {
281         int defaultFileMode = fileModes[0];
282         int oneFileMode = fileModes[1];
283         int twoFileMode = fileModes[2];
284 
285         TarArchiver archiver = getPosixTarArchiver();
286 
287         archiver.setDirectoryMode( directoryMode );
288 
289         archiver.setFileMode( defaultFileMode );
290 
291         archiver.addDirectory( getTestFile( "src/main" ) );
292         archiver.setFileMode( oneFileMode );
293 
294         archiver.addFile( getTestFile( "src/test/resources/manifests/manifest1.mf" ), "one.txt" );
295         archiver.addFile( getTestFile( "src/test/resources/manifests/manifest2.mf" ), "two.txt", twoFileMode );
296         archiver.setDestFile( getTestFile( "target/output/archive.tar" ) );
297 
298         archiver.addSymlink("link_to_test_destinaton", "../test_destination/");
299 
300         archiver.createArchive();
301 
302         TarArchiveInputStream tis;
303 
304         tis = new TarArchiveInputStream( bufferedInputStream( new FileInputStream( archiver.getDestFile() ) ) );
305         TarArchiveEntry te;
306 
307         while ( ( te = tis.getNextTarEntry() ) != null )
308         {
309             if ( te.isDirectory() )
310             {
311                 assertEquals( "un-expected tar-entry mode for [te.name=" + te.getName() + "]", directoryMode,
312                               te.getMode() & UnixStat.PERM_MASK );
313             }
314             else if ( te.isSymbolicLink() )
315             {
316                 assertEquals( "../test_destination/", te.getLinkName() );
317                 assertEquals( "link_to_test_destinaton", te.getName() );
318                 assertEquals( 0640, te.getMode() & UnixStat.PERM_MASK );
319             }
320             else
321             {
322                 if ( te.getName().equals( "one.txt" ) )
323                 {
324                     assertEquals( oneFileMode, te.getMode() & UnixStat.PERM_MASK );
325                 }
326                 else if ( te.getName().equals( "two.txt" ) )
327                 {
328                     assertEquals( twoFileMode, te.getMode() & UnixStat.PERM_MASK );
329                 }
330                 else
331                 {
332                     assertEquals( "un-expected tar-entry mode for [te.name=" + te.getName() + "]", defaultFileMode,
333                                   te.getMode() & UnixStat.PERM_MASK );
334                 }
335 
336             }
337         }
338         IOUtil.close( tis );
339 
340     }
