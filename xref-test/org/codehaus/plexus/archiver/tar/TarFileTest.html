90      private void testTarFile( Compressor compressor, String extension,
91                                TarFileCreator tarFileCreator )
92          throws Exception
93      {
94          File file = new File( "target/output/TarFileTest.tar" );
95          final TarArchiver archiver = (TarArchiver) lookup( Archiver.ROLE, "tar" );
96          archiver.setLongfile( TarLongFileMode.posix );
97          archiver.setDestFile( file );
98          archiver.addDirectory( new File( "src" ) );
99          FileUtils.removePath( file.getPath() );
100         archiver.createArchive();
101         if ( compressor != null )
102         {
103             final File compressedFile = new File( file.getPath() + extension );
104             compressor.setSource( createResource( file, file.getName() ) );
105             compressor.setDestFile( compressedFile );
106             compressor.compress();
107             compressor.close();
108             file = compressedFile;
109         }
110         final TarFile tarFile = tarFileCreator.newTarFile( file );
111 
112         for ( Enumeration en = tarFile.getEntries(); en.hasMoreElements(); )
113         {
114             final TarArchiveEntry te = (TarArchiveEntry) en.nextElement();
115             if ( te.isDirectory() || te.isSymbolicLink() )
116             {
117                 continue;
118             }
119             final File teFile = new File( "src", te.getName() );
120             final InputStream teStream = tarFile.getInputStream( te );
121             final InputStream fileStream = Files.newInputStream( teFile.toPath() );
122             assertTrue( Arrays.equals( IOUtil.toByteArray( teStream ), IOUtil.toByteArray( fileStream ) ) );
123             teStream.close();
124             fileStream.close();
125         }
126 
127         tarFile.close();
128     }
