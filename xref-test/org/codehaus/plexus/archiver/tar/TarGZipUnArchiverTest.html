24  package org.codehaus.plexus.archiver.tar;
25  
26  import java.io.File;
27  import org.codehaus.plexus.archiver.Archiver;
28  import org.codehaus.plexus.archiver.TestSupport;
29  import org.codehaus.plexus.archiver.UnArchiver;
30  import org.codehaus.plexus.archiver.gzip.GZipArchiver;
31  import org.junit.jupiter.api.Test;
32  
33  import static org.junit.jupiter.api.Assertions.assertEquals;
34  import static org.junit.jupiter.api.Assertions.assertTrue;
35  
36  /**
37   * @author Dan Tran
38   */
39  public class TarGZipUnArchiverTest
40          extends TestSupport
41  {
42  
43      @Test
44      public void testExtract()
45          throws Exception
46      {
47          TarArchiver tarArchiver = (TarArchiver) lookup( Archiver.class, "tar" );
48          tarArchiver.setLongfile( TarLongFileMode.posix );
49  
50          String fileName1 = "TarGZipUnArchiverTest1.txt";
51          String fileName2 = "TarGZipUnArchiverTest2.txt";
52          File file1InTar = getTestFile( "target/output/" + fileName1 );
53          File file2InTar = getTestFile( "target/output/" + fileName2 );
54          file1InTar.delete();
55          file2InTar.delete();
56  
57          tarArchiver.addFile( getTestFile( "src/test/resources/manifests/manifest1.mf" ), fileName1 );
58          tarArchiver.addFile( getTestFile( "src/test/resources/manifests/manifest2.mf" ), fileName2, 0664 );
59          tarArchiver.setDestFile( getTestFile( "target/output/archive.tar" ) );
60          tarArchiver.createArchive();
61  
62          GZipArchiver gzipArchiver = (GZipArchiver) lookup( Archiver.class, "gzip" );
63  
64          File testGZipFile = getTestFile( "target/output/archive.tar.gz" );
65          gzipArchiver.setDestFile( testGZipFile );
66          gzipArchiver.addFile( getTestFile( "target/output/archive.tar" ), "dontcare" );
67          gzipArchiver.createArchive();
68  
69          TarGZipUnArchiver tarGZipUnArchiver = (TarGZipUnArchiver) lookup( UnArchiver.class, "tgz" );
70          tarGZipUnArchiver.setDestDirectory( getTestFile( "target/output" ) );
71          tarGZipUnArchiver.setSourceFile( testGZipFile );
72          tarGZipUnArchiver.extract();
73  
74          assertTrue( file1InTar.exists() );
75          assertTrue( file2InTar.exists() );
76  
77          //make sure we place the source file back
78          assertEquals( testGZipFile, tarGZipUnArchiver.getSourceFile() );
79  
80      }
81  
82      @Test
83      public void testLookup()
84          throws Exception
85      {
86          lookup( UnArchiver.class, "tar.gz" );
87      }
88  
89  }
