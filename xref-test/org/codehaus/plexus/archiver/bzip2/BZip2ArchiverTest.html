63      public void testCreateResourceCollection()
64          throws Exception
65      {
66          final File pomFile = new File("pom.xml");
67          final File bz2File = new File( "target/output/pom.xml.bz2" );
68          BZip2Archiver bzip2Archiver = (BZip2Archiver) lookup( Archiver.ROLE, "bzip2" );
69          bzip2Archiver.setDestFile( bz2File );
70          bzip2Archiver.addFile( pomFile, "pom.xml" );
71          FileUtils.removePath( bz2File.getPath() );
72          bzip2Archiver.createArchive();
73          
74          System.out.println( "Created: " + bz2File.getAbsolutePath() );
75  
76          final File zipFile = new File( "target/output/pom.zip" );
77          ZipArchiver zipArchiver = (ZipArchiver) lookup( Archiver.ROLE, "zip" );
78          zipArchiver.setDestFile( zipFile );
79          zipArchiver.addArchivedFileSet( bz2File, "prfx/" );
80          FileUtils.removePath( zipFile.getPath() );
81          zipArchiver.createArchive();
82  
83          final ZipFile juZipFile = new ZipFile( zipFile );
84          final ZipEntry zipEntry = juZipFile.getEntry( "prfx/target/output/pom.xml" );
85          final InputStream archivePom = juZipFile.getInputStream( zipEntry );
86          final InputStream pom = new FileInputStream( pomFile );
87          
88          assertTrue( Arrays.equals( IOUtil.toByteArray( pom ), IOUtil.toByteArray( archivePom ) ) );
89          archivePom.close();
90          pom.close();
91          juZipFile.close();
92      }

94      /**
95       * Tests the .bzip2 archiver is forced set to true, and after that
96       * tests the behavior when the forced is set to false.
97       * 
98       * @throws Exception
99       */
100     public void testBz2IsForcedBehaviour() throws Exception
101     {
102         BZip2Archiver bZip2Archiver = (BZip2Archiver) createArchiver( "bzip2" );
103         
104         assertTrue( bZip2Archiver.isSupportingForced() );
105         bZip2Archiver.createArchive();
106         
107         final long creationTime = bZip2Archiver.getDestFile().lastModified();
108         
109         waitUntilNewTimestamp( bZip2Archiver.getDestFile(), creationTime );
110        
111         bZip2Archiver = (BZip2Archiver) createArchiver( "bzip2" );
112 
113         bZip2Archiver.setForced( true );
114         bZip2Archiver.createArchive();
115         
116         final long firstRunTime = bZip2Archiver.getDestFile().lastModified();
117 
118         assertFalse( creationTime==firstRunTime );
119         
120         bZip2Archiver = (BZip2Archiver) createArchiver( "bzip2" );
121 
122         bZip2Archiver.setForced( false );
123         bZip2Archiver.createArchive();
124         
125         final long secondRunTime = bZip2Archiver.getDestFile().lastModified();
126         
127         assertEquals( firstRunTime,secondRunTime );
128     }
