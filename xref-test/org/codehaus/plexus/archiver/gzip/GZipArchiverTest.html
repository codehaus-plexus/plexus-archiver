53      @Test
54      public void testCreateArchive()
55          throws Exception
56      {
57          ZipArchiver zipArchiver = (ZipArchiver) lookup( Archiver.class, "zip" );
58          zipArchiver.addDirectory( getTestFile( "src" ) );
59          zipArchiver.setDestFile( getTestFile( "target/output/archiveForGzip.zip" ) );
60          zipArchiver.createArchive();
61          GZipArchiver archiver = (GZipArchiver) lookup( Archiver.class, "gzip" );
62          String[] inputFiles = new String[ 1 ];
63          inputFiles[0] = "archiveForGzip.zip";
64          archiver.addDirectory( getTestFile( "target/output" ), inputFiles, null );
65          archiver.setDestFile( getTestFile( "target/output/archive.gzip" ) );
66          archiver.createArchive();
67      }

86      @Test
87      public void testCreateResourceCollection()
88          throws Exception
89      {
90          final File pomFile = new File( "pom.xml" );
91          final File gzFile = new File( "target/output/pom.xml.gz" );
92          GZipArchiver gzipArchiver = (GZipArchiver) lookup( Archiver.class, "gzip" );
93          gzipArchiver.setDestFile( gzFile );
94          gzipArchiver.addFile( pomFile, "pom.xml" );
95          FileUtils.removePath( gzFile.getPath() );
96          gzipArchiver.createArchive();
97  
98          final File zipFile = new File( "target/output/pom.zip" );
99          ZipArchiver zipArchiver = (ZipArchiver) lookup( Archiver.class, "zip" );
100         zipArchiver.setDestFile( zipFile );
101         zipArchiver.addArchivedFileSet( gzFile, "prfx/" );
102         FileUtils.removePath( zipFile.getPath() );
103         zipArchiver.createArchive();
104 
105         final ZipFile juZipFile = new ZipFile( zipFile );
106         final ZipEntry zipEntry = juZipFile.getEntry( "prfx/target/output/pom.xml" );
107         final InputStream archivePom = juZipFile.getInputStream( zipEntry );
108         final InputStream pom = Files.newInputStream( pomFile.toPath() );
109         assertTrue( Arrays.equals( IOUtil.toByteArray( pom ), IOUtil.toByteArray( archivePom ) ) );
110         archivePom.close();
111         pom.close();
112         juZipFile.close();
113     }

115     /**
116      * Tests the .gzip archiver is forced set to true, and after that
117      * tests the behavior when the forced is set to false.
118      *
119      * @throws Exception
120      */
121     @Test
122     public void testTarGzIsForcedBehaviour() throws Exception
123     {
124         GZipArchiver gZipArchiver = (GZipArchiver) createArchiver( "gzip" );
125 
126         assertTrue( gZipArchiver.isSupportingForced() );
127         gZipArchiver.createArchive();
128 
129         final long creationTime = gZipArchiver.getDestFile().lastModified();
130 
131         waitUntilNewTimestamp( gZipArchiver.getDestFile(), creationTime );
132 
133         gZipArchiver = (GZipArchiver) createArchiver( "gzip" );
134 
135         gZipArchiver.setForced( true );
136         gZipArchiver.createArchive();
137 
138         final long firstRunTime = gZipArchiver.getDestFile().lastModified();
139 
140         assertNotEquals( creationTime, firstRunTime );
141 
142         gZipArchiver = (GZipArchiver) createArchiver( "gzip" );
143 
144         gZipArchiver.setForced( false );
145         gZipArchiver.createArchive();
146 
147         final long secondRunTime = gZipArchiver.getDestFile().lastModified();
148 
149         assertEquals( firstRunTime, secondRunTime );
150     }
