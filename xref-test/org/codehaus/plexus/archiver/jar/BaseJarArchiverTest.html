38  public abstract class BaseJarArchiverTest
39  {
40  
41      /*
42       * Verify that the JarArchiver implementation
43       * could create basic JAR file
44       */
45      @Test
46      public void testCreateJar()
47          throws IOException, ArchiverException
48      {
49          File jarFile = new File( "target/output/testJar.jar" );
50          jarFile.delete();
51  
52          JarArchiver archiver = getJarArchiver();
53          archiver.setDestFile( jarFile );
54          archiver.addDirectory( new File( "src/test/resources/java-classes" ) );
55  
56          archiver.createArchive();
57  
58          // verify that the JAR file is created and contains the expected files
59          try ( ZipFile resultingArchive = new ZipFile( jarFile ) )
60          {
61              // verify that the JAR file contains manifest directory and file
62              // and that those are the first two entries.
63              Enumeration<? extends ZipEntry> resultingEntries = resultingArchive.entries();
64              assertEquals( "META-INF/", resultingEntries.nextElement().getName() );
65              assertEquals( "META-INF/MANIFEST.MF", resultingEntries.nextElement().getName() );
66  
67              // verify the JAR contains the class and it is not corrupted
68              ZipEntry classFileEntry = resultingArchive.getEntry( "com/example/app/Main.class" );
69              InputStream resultingClassFile = resultingArchive.getInputStream( classFileEntry );
70              InputStream originalClassFile =
71                  Files.newInputStream( Paths.get( "src/test/resources/java-classes/com/example/app/Main.class" ) );
72  
73              assertTrue( IOUtil.contentEquals( originalClassFile, resultingClassFile ) );
74          }
75      }
76  
77      protected static long normalizeLastModifiedTime( long dosTime )
78      {
79          Calendar cal = Calendar.getInstance( TimeZone.getDefault(), Locale.ROOT );
80          cal.setTimeInMillis( dosTime );
81          return dosTime - ( cal.get( Calendar.ZONE_OFFSET ) + cal.get( Calendar.DST_OFFSET ) );
82      }
83  
84      protected abstract JarArchiver getJarArchiver();

86  }
