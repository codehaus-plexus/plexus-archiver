49      @Test
50      public void testVeryLargeJar()
51          throws IOException, ManifestException, ArchiverException
52      {
53          // Generate some number of random files that is likely to be
54          // two or three times the number of available file handles
55          File tmpDir = File.createTempFile( "veryLargeJar", null );
56          tmpDir.delete();
57          tmpDir.mkdirs();
58          Random rand = new Random();
59          for ( int i = 0; i < 45000; i++ )
60          {
61              File f = new File( tmpDir, "file" + i );
62              f.deleteOnExit();
63              OutputStream out = Files.newOutputStream( f.toPath() );
64              byte[] data = new byte[ 512 ]; // 512bytes per file
65              rand.nextBytes( data );
66              out.write( data );
67              out.flush();
68              out.close();
69          }
70  
71          File jarFile = new File( "target/output/veryLargeJar.jar" );
72  
73          JarArchiver archiver = getJarArchiver();
74          archiver.setDestFile( jarFile );
75          archiver.addDirectory( tmpDir );
76          archiver.createArchive();
77      }
