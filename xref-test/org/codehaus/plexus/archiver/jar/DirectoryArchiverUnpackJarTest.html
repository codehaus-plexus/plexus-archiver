23      {
24          "**/*"

32          {
33          }
34  
35          @Nonnull
36          @Override
37          public InputStream transform( @Nonnull PlexusIoResource resource, @Nonnull InputStream inputStream )
38              throws IOException
39          {
40              return inputStream;
41          }

45      @Test
46      public void test_dependency_sets_depSet_unpacked_rdonly()
47          throws Exception
48      {
49          File src = new File( "src/test/resources/unpack_issue.jar" );
50          assertTrue( src.exists() );
51          DefaultArchivedFileSet afs = DefaultArchivedFileSet.archivedFileSet( src );
52          afs.setIncludes( DEFAULT_INCLUDES_ARRAY );
53          afs.setExcludes( null );
54          afs.setPrefix( "child-1/" );
55          afs.setStreamTransformer( new IdentityTransformer() );
56          Archiver archiver = (Archiver) lookup( Archiver.class, "dir" );
57          archiver.setDefaultDirectoryMode( 0555 );
58          archiver.setDirectoryMode( 0555 ); // causes permission denied if bug is not fixed.
59          archiver.setDestFile( new File( "target/depset_unpack" ) );
60          archiver.addArchivedFileSet( afs, Charset.forName( "UTF-8" ) );
61          archiver.createArchive();
62          assertTrue( new File( "target/depset_unpack/child-1/META-INF/MANIFEST.MF" ).exists() );
63  
64          // make them writeable or mvn clean will fail
65          ArchiveEntryUtils.chmod( new File( "target/depset_unpack/child-1/META-INF" ), 0777 );
66          ArchiveEntryUtils.chmod( new File( "target/depset_unpack/child-1/META-INF/maven" ), 0777 );
67          ArchiveEntryUtils.chmod( new File( "target/depset_unpack/child-1/META-INF/maven/test" ), 0777 );
68          ArchiveEntryUtils.chmod( new File( "target/depset_unpack/child-1/META-INF/maven/test/child1" ), 0777 );
69          ArchiveEntryUtils.chmod( new File( "target/depset_unpack/child-1/assembly-resources" ), 0777 );
70      }
