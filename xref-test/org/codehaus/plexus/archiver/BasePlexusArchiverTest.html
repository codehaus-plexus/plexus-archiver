24  package org.codehaus.plexus.archiver;
25  
26  import java.io.File;
27  import java.io.IOException;
28  import java.nio.file.Files;
29  import java.nio.file.attribute.FileTime;
30  
31  import org.codehaus.plexus.util.FileUtils;
32  
33  /**
34   * Base abstract class that all the test-cases for different archivers
35   * extend so that they can use its helpful methods.
36   */
37  public abstract class BasePlexusArchiverTest extends TestSupport
38  {
39  
40      /**
41       * Ensure that the last modified timestamp of a file will be greater
42       * than the one specified as a reference.
43       *
44       * @param outputFile the file
45       * @param timestampReference the file will have a newer timestamp
46       *        than this reference timestamp.
47       *
48       * @throws IOException if the timestamp could not be modified
49       */
50      protected void waitUntilNewTimestamp( File outputFile, long timestampReference )
51          throws IOException
52      {
53          long startTime = System.currentTimeMillis();
54          File tmpFile = File.createTempFile(
55              "BasePlexusArchiverTest.waitUntilNewTimestamp", null );
56          long newTimestamp;
57  
58          // We could easily just set the last modified time using
59          // Files.setLastModifiedTime and System.currentTimeMillis(),
60          // but the problem is that tests are using this method to verify that
61          // the force flag is working. To ensure that modified or
62          // newly created files will have timestamp newer than
63          // `timestampReference`, we need to modify a file ourself.
64          // Otherwise the build may fail because when the test overrides
65          // `outputFile` it will have timestamp that is equal
66          // to `timestampReference`.
67          do
68          {
69              FileUtils.fileWrite( tmpFile, "waitUntilNewTimestamp" );
70              newTimestamp = tmpFile.lastModified();
71              Thread.yield();
72          }
73          while ( timestampReference >= newTimestamp
74                  // A simple guard to ensure that we'll not do this forever.
75                  // If the last modified timestamp is not changed to
76                  // a newer value after 10 seconds, probably it never will.
77                  && System.currentTimeMillis() - startTime < 10_000 );
78  
79          tmpFile.delete();
80  
81          if ( timestampReference >= newTimestamp )
82          {
83              throw new IOException("Could not modify the last modified timestamp "
84                  + "to newer than the refence value." );
85          }
86  
87          FileTime newTimestampTime = FileTime.fromMillis( newTimestamp );
88          Files.setLastModifiedTime( outputFile.toPath(), newTimestampTime );
89      }
90  
91      /**
92       * Base method for all the Archivers to create an archiver.
93       *
94       * @param format
95       *
96       * @return
97       *
98       * @throws Exception
99       */
100     protected Archiver createArchiver( String format ) throws Exception
101     {
102 
103         final File pomFile = new File( "pom.xml" );
104         final File rarFile = new File( "target/output/pom.xml." + format );
105 
106         Archiver archiver = (Archiver) lookup( Archiver.class, format );
107         archiver.setDestFile( rarFile );
108         archiver.addFile( pomFile, "pom.xml" );
109 
110         return archiver;
111     }
112 
113 }
