11  /**
12   * Test support class.
13   */
14  public abstract class TestSupport
15          extends InjectedTest
16  {
17      @Override
18      @BeforeEach
19      public void setUp() throws Exception {
20          super.setUp();
21      }
22  
23      @Override
24      @AfterEach
25      public void tearDown() throws Exception {
26          super.tearDown();
27      }
28  
29      private static Path basedir()
30      {
31          return Paths.get( System.getProperty( "basedir", ( new File( "" ) ).getAbsolutePath() ) );
32      }
33  
34      protected static File getTestFile( final String path )
35      {
36          return basedir().resolve( path ).toFile();
37      }
38  
39      protected static File getTestFile( final String basedir, final String path )
40      {
41          return basedir().resolve( basedir ).resolve( path ).toFile();
42      }
43  }
