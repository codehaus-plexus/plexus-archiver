6   import java.io.File;

8   import java.io.InputStream;
9   import java.util.HashMap;
10  import java.util.zip.GZIPInputStream;
11  
12  import javax.annotation.Nonnull;
13  import javax.inject.Named;
14  
15  import org.codehaus.plexus.components.io.attributes.FileAttributes;
16  import org.codehaus.plexus.components.io.attributes.PlexusIoResourceAttributes;
17  import org.codehaus.plexus.components.io.resources.PlexusIoCompressedFileResourceCollection;
18  
19  /**
20   * Abstract base class for compressed files, aka singleton
21   * resource collections.
22   */
23  @Named( "gzip" )
24  public class PlexusIoGzipResourceCollection
25      extends PlexusIoCompressedFileResourceCollection
26  {
27  
28      @Override
29      protected String getDefaultExtension()
30      {
31          return ".gz";
32      }
33  
34      @Nonnull
35      @Override
36      protected InputStream getInputStream( File file )
37          throws IOException
38      {
39          return bufferedInputStream( new GZIPInputStream( fileInputStream( file ) ) );
40      }
41  
42      @Override
43      protected PlexusIoResourceAttributes getAttributes( File file )
44          throws IOException
45      {
46          return new FileAttributes( file, new HashMap<Integer, String>(), new HashMap<Integer, String>() );
47      }
48  
49  }
