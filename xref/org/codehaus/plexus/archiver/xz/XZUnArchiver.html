16  package org.codehaus.plexus.archiver.xz;

18  import java.io.File;
19  import java.io.IOException;
20  import java.io.InputStream;
21  import javax.annotation.Nonnull;
22  import org.apache.commons.compress.compressors.xz.XZCompressorInputStream;
23  import org.codehaus.plexus.archiver.AbstractUnArchiver;
24  import org.codehaus.plexus.archiver.ArchiverException;
25  
26  import static org.codehaus.plexus.archiver.util.Streams.bufferedInputStream;
27  import static org.codehaus.plexus.archiver.util.Streams.bufferedOutputStream;
28  import static org.codehaus.plexus.archiver.util.Streams.copyFully;
29  import static org.codehaus.plexus.archiver.util.Streams.fileInputStream;
30  import static org.codehaus.plexus.archiver.util.Streams.fileOutputStream;
31  
32  /**
33   *
34   * @author philip.lourandos
35   * @since 3.3
36   */
37  public class XZUnArchiver extends AbstractUnArchiver 
38  {
39      private static final String OPERATION_XZ = "xz";
40  
41      public XZUnArchiver()
42      {
43      }
44      
45      public XZUnArchiver( File source )
46      {
47          super( source );
48      }
49  
50      @Override
51      protected void execute() throws ArchiverException
52      {
53          if ( getSourceFile().lastModified() > getDestFile().lastModified() )
54          {
55              getLogger().info( "Expanding " + getSourceFile().getAbsolutePath() + " to " 
56                  + getDestFile().getAbsolutePath() );
57              
58              copyFully( getXZInputStream( bufferedInputStream( fileInputStream( getSourceFile(), OPERATION_XZ) ) ),
59                      bufferedOutputStream( fileOutputStream( getDestFile(), OPERATION_XZ) ), OPERATION_XZ );
60          }
61      }
62  
63      public static @Nonnull XZCompressorInputStream getXZInputStream( InputStream in )
64              throws ArchiverException
65      {
66          try 
67          {
68              return new XZCompressorInputStream(in);
69          }
70          catch (IOException ioe) 
71          {
72              throw new ArchiverException( "Trouble creating BZIP2 compressor, invalid file ?", ioe );
73          }
74      }
75      
76      @Override
77      protected void execute(String path, File outputDirectory) throws ArchiverException {
78          throw new UnsupportedOperationException("Targeted execution not supported in xz format");
79          
80      }
81  }
