94      protected void execute( File sourceFile, File destDirectory )
95          throws ArchiverException
96      {
97          TarArchiveInputStream tis = null;
98          try
99          {
100             getLogger().info( "Expanding: " + sourceFile + " into " + destDirectory );
101             TarFile tarFile = new TarFile( sourceFile );
102             tis = new TarArchiveInputStream(
103                 decompress( compression, sourceFile, new BufferedInputStream( new FileInputStream( sourceFile ) ) ) );
104             TarArchiveEntry te;
105             while ( ( te = tis.getNextTarEntry() ) != null )
106             {
107                 TarResource fileInfo = new TarResource( tarFile, te );
108                 if ( isSelected( te.getName(), fileInfo ) )
109                 {
110                     final String symlinkDestination = te.isSymbolicLink() ? te.getLinkName() : null;
111                     extractFile( sourceFile, destDirectory, tis, te.getName(), te.getModTime(), te.isDirectory(),
112                                  te.getMode() != 0 ? te.getMode() : null, symlinkDestination );
113                 }
114 
115             }
116             getLogger().debug( "expand complete" );
117 
118         }
119         catch ( IOException ioe )
120         {
121             throw new ArchiverException( "Error while expanding " + sourceFile.getAbsolutePath(), ioe );
122         }
123         finally
124         {
125             IOUtil.close( tis );
126         }
127     }

129     /**
130      * This method wraps the input stream with the
131      * corresponding decompression method
132      *
133      * @param file    provides location information for BuildException
134      * @param istream input stream
135      * @return input stream with on-the-fly decompression
136      * @throws IOException thrown by GZIPInputStream constructor
137      */
138     private InputStream decompress( UntarCompressionMethod compression, final File file, final InputStream istream )
139         throws IOException, ArchiverException
140     {
141         if ( compression == UntarCompressionMethod.GZIP )
142         {
143             return Streams.bufferedInputStream( new GZIPInputStream( istream ) );
144         }
145         else if ( compression == UntarCompressionMethod.BZIP2 )
146         {
147             return new BZip2CompressorInputStream( istream );
148         }
149         else if ( compression == UntarCompressionMethod.SNAPPY )
150         {
151             return new SnappyInputStream( istream, true );
152         }
153         return istream;
154     }
