18  /**
19   * <p>
20   * Implementation of {@link ArchiveFile} for tar files.</p>
21   * <p>
22   * Compared to
23   * {@link org.apache.commons.compress.archivers.zip.ZipFile}, this one should be used with some care, due to the
24   * nature of a tar file: While a zip file contains a catalog, a tar
25   * file does not. In other words, the only way to read a tar file in
26   * a performant manner is by iterating over it from the beginning to
27   * the end. If you try to open another entry than the "next" entry,
28   * then you force to skip entries, until the requested entry is found.
29   * This may require to reread the entire file!</p>
30   * <p>
31   * In other words, the recommended use of this class is to use
32   * {@link #getEntries()} and invoke {@link #getInputStream(TarArchiveEntry)}
33   * only for the current entry. Basically, this is to handle it like
34   * {@link TarArchiveInputStream}.</p>
35   * <p>
36   * The advantage of this class is that you may write code for the
37   * {@link ArchiveFile}, which is valid for both tar files and zip files.</p>
38   */
39  public class TarFile
40      implements ArchiveFile
41  {
42  
43      private final java.io.File file;
44  
45      private TarArchiveInputStream inputStream;
46  
47      private TarArchiveEntry currentEntry;
48  
49      /**
50       * Creates a new instance with the given file.
51       */
52      public TarFile( File file )
53      {
54          this.file = file;
55      }
56  
57      /**
58       * Implementation of {@link ArchiveFile#getEntries()}. Note, that there is
59       * an interaction between this method and {@link #getInputStream(TarArchiveEntry)},
60       * or {@link #getInputStream(org.apache.commons.compress.archivers.ArchiveEntry)}:
61       * If an input stream is opened for any other entry than the enumerations
62       * current entry, then entries may be skipped.
63       */
64      @Override
65      public Enumeration<org.apache.commons.compress.archivers.ArchiveEntry> getEntries()
66          throws IOException
67      {
68          if ( inputStream != null )
69          {
70              close();
71          }
72          open();
73          return new Enumeration<org.apache.commons.compress.archivers.ArchiveEntry>()
74          {
75  
76              boolean currentEntryValid;
77  
78              @Override
79              public boolean hasMoreElements()
80              {
81                  if ( !currentEntryValid )
82                  {
83                      try
84                      {
85                          currentEntry = inputStream.getNextTarEntry();
86                      }
87                      catch ( IOException e )
88                      {
89                          throw new UndeclaredThrowableException( e );
90                      }
91                  }
92                  return currentEntry != null;
93              }
94  
95              @Override
96              public org.apache.commons.compress.archivers.ArchiveEntry nextElement()
97              {
98                  if ( currentEntry == null )
99                  {
100                     throw new NoSuchElementException();
101                 }
102                 currentEntryValid = false;
103                 return currentEntry;
104             }
105 
106         };
107     }
108 
109     public void close()
110         throws IOException
111     {
112         if ( inputStream != null )
113         {
114             inputStream.close();
115             inputStream = null;
116         }
117     }
118 
119     @Override
120     public InputStream getInputStream( org.apache.commons.compress.archivers.ArchiveEntry entry )
121         throws IOException
122     {
123         return getInputStream( new TarArchiveEntry( entry.getName() ) );
124     }
125 
126     /**
127      * Returns an {@link InputStream} with the given entries
128      * contents. This {@link InputStream} may be closed: Nothing
129      * happens in that case, because an actual close would invalidate
130      * the underlying {@link TarArchiveInputStream}.
131      */
132     public InputStream getInputStream( TarArchiveEntry entry )
133         throws IOException
134     {
135         if ( entry.equals( (Object) currentEntry ) && inputStream != null )
136         {
137             return new FilterInputStream( inputStream )
138             {
139 
140                 public void close()
141                     throws IOException
142                 {
143                     // Does nothing.
144                 }
145 
146             };
147         }
148         return getInputStream( entry, currentEntry );
149     }
150 
151     protected InputStream getInputStream( File file )
152         throws IOException
153     {
154         return Streams.fileInputStream( file );
155     }
156 
157     private InputStream getInputStream( TarArchiveEntry entry, TarArchiveEntry currentEntry )
158         throws IOException
159     {
160         if ( currentEntry == null || inputStream == null )
161         {
162             // Search for the entry from the beginning of the file to the end.
163             if ( inputStream != null )
164             {
165                 close();
166             }
167             open();
168             if ( !findEntry( entry, null ) )
169             {
170                 throw new IOException( "Unknown entry: " + entry.getName() );
171             }
172         }
173         else
174         {
175             // Search for the entry from the current position to the end of the file.
176             if ( findEntry( entry, null ) )
177             {
178                 return getInputStream( entry );
179             }
180             close();
181             open();
182             if ( !findEntry( entry, currentEntry ) )
183             {
184                 throw new IOException( "No such entry: " + entry.getName() );
185             }
186         }
187         return getInputStream( entry );
188     }
189 
190     private void open()
191         throws IOException
192     {
193         inputStream = new TarArchiveInputStream( bufferedInputStream( getInputStream( file ) ), "UTF8" );
194     }
195 
196     private boolean findEntry( TarArchiveEntry entry, TarArchiveEntry currentEntry )
197         throws IOException
198     {
199         for ( ;; )
200         {
201             this.currentEntry = inputStream.getNextTarEntry();
202             if ( this.currentEntry == null
203                      || ( currentEntry != null && this.currentEntry.equals( currentEntry ) ) )
204             {
205                 return false;
206             }
207             if ( this.currentEntry.equals( entry ) )
208             {
209                 return true;
210             }
211         }
212     }
