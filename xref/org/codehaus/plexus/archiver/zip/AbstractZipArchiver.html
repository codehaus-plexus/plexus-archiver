103     /**
104      * Whether the file modification times will be rounded up to the
105      * next even number of seconds.
106      * <p/>
107      * <p>Zip archives store file modification times with a
108      * granularity of two seconds, so the times will either be rounded
109      * up or down.  If you round down, the archive will always seem
110      * out-of-date when you rerun the task, so the default is to round
111      * up.  Rounding up may lead to a different type of problems like
112      * JSPs inside a web archive that seem to be slightly more recent
113      * than precompiled pages, rendering precompilation useless.</p>	 *
114      * <p/>
115      * plexus-archiver chooses to round up.
116      * <p/>
117      * Java versions up to java7 round timestamp down, which means we add a heuristic value (which is slightly
118      * questionable)
119      * Java versions from 8 and up round timestamp up.
120      * s
121      */
122     private static final boolean isJava7OrLower =
123         Integer.parseInt( System.getProperty( "java.version" ).split( "\\." )[1] ) <= 7;

204     protected void execute()
205         throws ArchiverException, IOException
206     {
207         if ( !checkForced() )
208         {
209             return;
210         }
211 
212         if ( doubleFilePass )
213         {
214             skipWriting = true;
215             createArchiveMain();
216             skipWriting = false;
217             createArchiveMain();
218         }
219         else
220         {
221             createArchiveMain();
222         }
223 
224         finalizeZipOutputStream( zOut );
225     }

232     private void createArchiveMain()
233         throws ArchiverException, IOException
234     {
235         //noinspection deprecation
236         if ( !Archiver.DUPLICATES_SKIP.equals( duplicate ) )
237         {
238             //noinspection deprecation
239             setDuplicateBehavior( duplicate );
240         }
241 
242         ResourceIterator iter = getResources();
243         if ( !iter.hasNext() && !hasVirtualFiles() )
244         {
245             throw new ArchiverException( "You must set at least one file." );
246         }
247 
248         zipFile = getDestFile();
249 
250         if ( zipFile == null )
251         {
252             throw new ArchiverException( "You must set the destination " + archiveType + "file." );
253         }
254 
255         if ( zipFile.exists() && !zipFile.isFile() )
256         {
257             throw new ArchiverException( zipFile + " isn't a file." );
258         }
259 
260         if ( zipFile.exists() && !zipFile.canWrite() )
261         {
262             throw new ArchiverException( zipFile + " is read-only." );
263         }
264 
265         // Whether or not an actual update is required -
266         // we don't need to update if the original file doesn't exist
267 
268         addingNewFiles = true;
269 
270         if ( doUpdate && !zipFile.exists() )
271         {
272             doUpdate = false;
273             getLogger().debug( "ignoring update attribute as " + archiveType + " doesn't exist." );
274         }
275 
276         success = false;
277 
278         if ( doUpdate )
279         {
280             renamedFile = FileUtils.createTempFile( "zip", ".tmp", zipFile.getParentFile() );
281             renamedFile.deleteOnExit();
282 
283             try
284             {
285                 FileUtils.rename( zipFile, renamedFile );
286             }
287             catch ( SecurityException e )
288             {
289                 getLogger().debug( e.toString() );
290                 throw new ArchiverException(
291                     "Not allowed to rename old file (" + zipFile.getAbsolutePath() + ") to temporary file", e );
292             }
293             catch ( IOException e )
294             {
295                 getLogger().debug( e.toString() );
296                 throw new ArchiverException(
297                     "Unable to rename old file (" + zipFile.getAbsolutePath() + ") to temporary file", e );
298             }
299         }
300 
301         String action = doUpdate ? "Updating " : "Building ";
302 
303         getLogger().info( action + archiveType + ": " + zipFile.getAbsolutePath() );
304 
305         if ( !skipWriting )
306         {
307             zipArchiveOutputStream =
308                 new ZipArchiveOutputStream( bufferedOutputStream( fileOutputStream( zipFile, "zip" ) ) );
309             zipArchiveOutputStream.setEncoding( encoding );
310             zipArchiveOutputStream.setCreateUnicodeExtraFields(
311                 ZipArchiveOutputStream.UnicodeExtraFieldPolicy.NOT_ENCODEABLE );
312             zipArchiveOutputStream.setMethod(
313                 doCompress ? ZipArchiveOutputStream.DEFLATED : ZipArchiveOutputStream.STORED );
314 
315             zOut = new ConcurrentJarCreator(Runtime.getRuntime().availableProcessors());
316         }
317         initZipOutputStream( zOut );
318 
319         // Add the new files to the archive.
320         addResources( iter, zOut );
321 
322         // If we've been successful on an update, delete the
323         // temporary file
324         if ( doUpdate )
325         {
326             if ( !renamedFile.delete() )
327             {
328                 getLogger().warn( "Warning: unable to delete temporary file " + renamedFile.getName() );
329             }
330         }
331         success = true;
332     }

334     /**
335      * Add the given resources.
336      *
337      * @param resources the resources to add
338      * @param zOut      the stream to write to
339      */
340     @SuppressWarnings( { "JavaDoc" } )
341     protected final void addResources( ResourceIterator resources, ConcurrentJarCreator zOut )
342         throws IOException, ArchiverException
343     {
344         while ( resources.hasNext() )
345         {
346             ArchiveEntry entry = resources.next();
347             String name = entry.getName();
348             name = name.replace( File.separatorChar, '/' );
349 
350             if ( "".equals( name ) )
351             {
352                 continue;
353             }
354 
355             if ( entry.getResource().isDirectory() && !name.endsWith( "/" ) )
356             {
357                 name = name + "/";
358             }
359 
360             addParentDirs( entry, null, name, zOut);
361 
362             if ( entry.getResource().isFile() )
363             {
364                 zipFile( entry, zOut, name );
365             }
366             else
367             {
368                 zipDir( entry.getResource(), zOut, name, entry.getMode(), encoding );
369             }
370         }
371     }

373     /**
374      * Ensure all parent dirs of a given entry have been added.
375      * <p/>
376      * This method is computed in terms of the potentially remapped entry (that may be disconnected from the file system)
377      * we do not *relly* know the entry's connection to the file system so establishing the attributes of the parents can
378      * be impossible and is not really supported.
379      */
380     @SuppressWarnings( { "JavaDoc" } )
381     private void addParentDirs(ArchiveEntry archiveEntry, File baseDir, String entry, ConcurrentJarCreator zOut)
382         throws IOException
383     {
384         if ( !doFilesonly && getIncludeEmptyDirs() )
385         {
386 			Stack<String> directories = addedDirs.asStringStack(entry);
387 
388             while ( !directories.isEmpty() )
389             {
390                 String dir = directories.pop();
391                 File f;
392                 if ( baseDir != null )
393                 {
394                     f = new File( baseDir, dir );
395                 }
396                 else
397                 {
398                     f = new File( dir );
399                 }
400                 // the
401                 // At this point we could do something like read the atr
402                 final PlexusIoResource res = new AnonymousResource( f );
403                 zipDir( res, zOut, dir, archiveEntry.getDefaultDirMode(), encoding );
404             }
405         }
406     }

408 	/**
409      * Adds a new entry to the archive, takes care of duplicates as well.
410      *  @param in                 the stream to read data for the entry from.
411      * @param zOut               the stream to write to.
412      * @param vPath              the name this entry shall have in the archive.
413      * @param lastModified       last modification time for the entry.
414      * @param fromArchive        the original archive we are copying this
415      * @param symlinkDestination
416      */
417     @SuppressWarnings( { "JavaDoc" } )
418     protected void zipFile( InputStreamSupplier in, ConcurrentJarCreator zOut, String vPath,
419                             long lastModified,
420                             File fromArchive, int mode, String symlinkDestination )
421         throws IOException, ArchiverException
422     {
423         getLogger().debug( "adding entry " + vPath );
424 
425         entries.put( vPath, vPath );
426 
427         if ( !skipWriting )
428         {
429             ZipArchiveEntry ze = new ZipArchiveEntry( vPath );
430             setTime( ze, lastModified );
431 
432             ze.setMethod( doCompress ? ZipArchiveEntry.DEFLATED : ZipArchiveEntry.STORED );
433             ze.setUnixMode( UnixStat.FILE_FLAG | mode );
434 
435             InputStream payload;
436             if ( ze.isUnixSymlink() )
437             {
438                 ZipEncoding enc = ZipEncodingHelper.getZipEncoding( getEncoding() );
439                 final byte[] bytes = enc.encode( symlinkDestination ).array();
440                 payload = new ByteArrayInputStream( bytes );
441                 zOut.addArchiveEntry( ze, createInputStreamSupplier( payload ) );
442             }
443             else
444             {
445                 zOut.addArchiveEntry( ze, wrappedRecompressor( ze, in ) );
446             }
447         }
448     }

455     private boolean isZipHeader( byte[] header )
456     {
457         return header[0] == 0x50 && header[1] == 0x4b && header[2] == 3 && header[3] == 4;
458     }

460     /**
461      * Method that gets called when adding from java.io.File instances.
462      * <p/>
463      * <p>This implementation delegates to the six-arg version.</p>
464      *  @param entry the file to add to the archive
465      * @param zOut  the stream to write to
466 	 * @param vPath the name this entry shall have in the archive
467 	 */
468     @SuppressWarnings( { "JavaDoc" } )
469     protected void zipFile( final ArchiveEntry entry, ConcurrentJarCreator zOut, String vPath )
470         throws IOException, ArchiverException
471     {
472         final PlexusIoResource resource = entry.getResource();
473         if ( ResourceUtils.isSame( resource, getDestFile() ) )
474         {
475             throw new ArchiverException( "A zip file cannot include itself" );
476         }
477 
478         final boolean b = entry.getResource() instanceof SymlinkDestinationSupplier;
479         String symlinkTarget = b ? ( (SymlinkDestinationSupplier) entry.getResource() ).getSymlinkDestination() : null;
480         InputStreamSupplier in = new InputStreamSupplier()
481         {
482             @Override
483             public InputStream get()
484             {
485                 try
486                 {
487                     return entry.getInputStream();
488                 }
489                 catch ( IOException e )
490                 {
491                     throw new RuntimeException( e );
492                 }
493             }
494         };
495         try
496         {
497             zipFile( in, zOut, vPath, resource.getLastModified(), null, entry.getMode(), symlinkTarget );
498         }
499         catch ( IOException e )
500         {
501             throw new ArchiverException( "IOException when zipping r" + entry.getName() + ": " + e.getMessage(), e );
502         }
503     }

523     protected void zipDir( PlexusIoResource dir, ConcurrentJarCreator zOut, String vPath, int mode,
524                            String encodingToUse )
525         throws IOException
526     {
527         if ( addedDirs.update(vPath)  )
528         {
529             return;
530         }
531 
532         getLogger().debug("adding directory " + vPath);
533 
534         if ( !skipWriting )
535         {
536             final boolean isSymlink = dir instanceof SymlinkDestinationSupplier;
537 
538             if ( isSymlink && vPath.endsWith( File.separator ) )
539             {
540                 vPath = vPath.substring( 0, vPath.length() - 1 );
541             }
542 
543             ZipArchiveEntry ze = new ZipArchiveEntry( vPath );
544 
545             /*
546              * ZipOutputStream.putNextEntry expects the ZipEntry to
547              * know its size and the CRC sum before you start writing
548              * the data when using STORED mode - unless it is seekable.
549              *
550              * This forces us to process the data twice.
551              */
552 
553             if ( isSymlink )
554             {
555                 mode = UnixStat.LINK_FLAG | mode;
556             }
557 
558             if ( dir != null && dir.isExisting() )
559             {
560                 setTime( ze, dir.getLastModified() );
561             }
562             else
563             {
564                 // ZIPs store time with a granularity of 2 seconds, round up
565                 setTime( ze, System.currentTimeMillis() );
566             }
567             if ( !isSymlink )
568             {
569                 ze.setSize( 0 );
570                 ze.setMethod( ZipArchiveEntry.STORED );
571                 // This is faintly ridiculous:
572                 ze.setCrc( EMPTY_CRC );
573             }
574             ze.setUnixMode( mode );
575 
576             if ( !isSymlink )
577             {
578                 zOut.addArchiveEntry( ze, createInputStreamSupplier( new ByteArrayInputStream( "".getBytes() ) ) );
579             }
580             else
581             {
582                 String symlinkDestination = ( (SymlinkDestinationSupplier) dir ).getSymlinkDestination();
583                 ZipEncoding enc = ZipEncodingHelper.getZipEncoding( encodingToUse );
584                 final byte[] bytes = enc.encode( symlinkDestination ).array();
585                 ze.setMethod( ZipArchiveEntry.DEFLATED );
586                 zOut.addArchiveEntry( ze, createInputStreamSupplier( new ByteArrayInputStream( bytes ) ) );
587             }
588         }
589     }

592     private InputStreamSupplier wrappedRecompressor( final ZipArchiveEntry ze, final InputStreamSupplier other )
593     {
594 
595         return new InputStreamSupplier()
596         {
597             public InputStream get()
598             {
599                 InputStream is = other.get();
600                 byte[] header = new byte[4];
601                 try
602                 {
603                     int read = is.read( header );
604                     boolean compressThis = doCompress;
605                     if ( !recompressAddedZips && isZipHeader( header ) )
606                     {
607                         compressThis = false;
608                     }
609 
610                     ze.setMethod( compressThis ? ZipArchiveEntry.DEFLATED : ZipArchiveEntry.STORED );
611 
612                     return maybeSequence( header, read, is );
613                 }
614                 catch ( IOException e )
615                 {
616                     throw new RuntimeException( e );
617                 }
618 
619             }
620         };
621     }

634     /**
635      * Create an empty zip file
636      *
637      * @param zipFile The file
638      * @return true for historic reasons
639      */
640     @SuppressWarnings( { "JavaDoc" } )
641     protected boolean createEmptyZip( File zipFile )
642         throws ArchiverException
643     {
644         // In this case using java.util.zip will not work
645         // because it does not permit a zero-entry archive.
646         // Must create it manually.
647         getLogger().info( "Note: creating empty " + archiveType + " archive " + zipFile );
648         OutputStream os = null;
649         try
650         {
651             os = new FileOutputStream( zipFile );
652             // Cf. PKZIP specification.
653             byte[] empty = new byte[22];
654             empty[0] = 80; // P
655             empty[1] = 75; // K
656             empty[2] = 5;
657             empty[3] = 6;
658             // remainder zeros
659             os.write( empty );
660         }
661         catch ( IOException ioe )
662         {
663             throw new ArchiverException( "Could not create empty ZIP archive " + "(" + ioe.getMessage() + ")", ioe );
664         }
665         finally
666         {
667             IOUtil.close( os );
668         }
669         return true;
670     }

735     protected boolean revert( StringBuffer messageBuffer )
736     {
737         int initLength = messageBuffer.length();
738 
739         // delete a bogus ZIP file (but only if it's not the original one)
740         if ( ( !doUpdate || renamedFile != null ) && !zipFile.delete() )
741         {
742             messageBuffer.append( " (and the archive is probably corrupt but I could not delete it)" );
743         }
744 
745         if ( doUpdate && renamedFile != null )
746         {
747             try
748             {
749                 FileUtils.rename( renamedFile, zipFile );
750             }
751             catch ( IOException e )
752             {
753                 messageBuffer.append( " (and I couldn't rename the temporary file " );
754                 messageBuffer.append( renamedFile.getName() );
755                 messageBuffer.append( " back)" );
756             }
757         }
758 
759         return messageBuffer.length() == initLength;
760     }

762     protected void close()
763         throws IOException
764     {
765         // Close the output stream.
766         try
767         {
768             if ( zipArchiveOutputStream != null )
769             {
770                 zOut.writeTo( zipArchiveOutputStream );
771                 zipArchiveOutputStream.close();
772             }
773         }
774         catch ( IOException ex )
775         {
776             // If we're in this finally clause because of an
777             // exception, we don't really care if there's an
778             // exception when closing the stream. E.g. if it
779             // throws "ZIP file must have at least one entry",
780             // because an exception happened before we added
781             // any files, then we must swallow this
782             // exception. Otherwise, the error that's reported
783             // will be the close() error, which is not the
784             // real cause of the problem.
785             if ( success )
786             {
787                 throw ex;
788             }
789 
790         }
791         catch ( InterruptedException e )
792         {
793             IOException ex = new IOException( "InterruptedException exception" );
794             ex.initCause( e.getCause() );
795             throw ex;
796         }
797         catch ( ExecutionException e )
