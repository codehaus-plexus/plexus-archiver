28  /**
29   * This class implements an output stream in which the data is
30   * written into a byte array. The buffer automatically grows as data
31   * is written to it.
32   * <p>
33   * The data can be retrieved using <code>toByteArray()</code> and
34   * <code>toString()</code>.
35   * <p>
36   * Closing a {@code ByteArrayOutputStream} has no effect. The methods in
37   * this class can be called after the stream has been closed without
38   * generating an {@code IOException}.
39   * <p>
40   * This is an alternative implementation of the {@link java.io.ByteArrayOutputStream}
41   * class. The original implementation only allocates 32 bytes at the beginning.
42   * As this class is designed for heavy duty it starts at 1024 bytes. In contrast
43   * to the original it doesn't reallocate the whole memory block but allocates
44   * additional buffers. This way no buffers need to be garbage collected and
45   * the contents don't have to be copied to the new buffer. This class is
46   * designed to behave exactly like the original. The only exception is the
47   * deprecated toString(int) method that has been ignored.
48   *
49   * @version $Id$
50   */
51  public class ByteArrayOutputStream extends OutputStream {
52  
53      /** A singleton empty byte array. */
54      private static final byte[] EMPTY_BYTE_ARRAY = new byte[0];
55  
56      /** The list of buffers, which grows and never reduces. */
57      private final List<byte[]> buffers = new ArrayList<byte[]>();
58      /** The index of the current buffer. */
59      private int currentBufferIndex;
60      /** The total count of bytes in all the filled buffers. */
61      private int filledBufferSum;
62      /** The current buffer. */
63      private byte[] currentBuffer;
64      /** The total count of bytes written. */
65      private int count;
66      /** Flag to indicate if the buffers can be reused after reset */
67      private boolean reuseBuffers = true;

69      /**
70       * Creates a new byte array output stream. The buffer capacity is
71       * initially 1024 bytes, though its size increases if necessary.
72       */
73      public ByteArrayOutputStream() {
74          this(1024);
75      }

77      /**
78       * Creates a new byte array output stream, with a buffer capacity of
79       * the specified size, in bytes.
80       *
81       * @param size  the initial size
82       * @throws IllegalArgumentException if size is negative
83       */
84      public ByteArrayOutputStream(final int size) {
85          if (size < 0) {
86              throw new IllegalArgumentException(
87                  "Negative initial size: " + size);
88          }
89          synchronized (this) {
90              needNewBuffer(size);
91          }
92      }
93  
94      /**
95       * Makes a new buffer available either by allocating
96       * a new one or re-cycling an existing one.
97       *
98       * @param newcount  the size of the buffer if one is created
99       */
100     private void needNewBuffer(final int newcount) {
101         if (currentBufferIndex < buffers.size() - 1) {
102             //Recycling old buffer
103             filledBufferSum += currentBuffer.length;
104 
105             currentBufferIndex++;
106             currentBuffer = buffers.get(currentBufferIndex);
107         } else {
108             //Creating new buffer
109             int newBufferSize;
110             if (currentBuffer == null) {
111                 newBufferSize = newcount;
112                 filledBufferSum = 0;
113             } else {
114                 newBufferSize = Math.max(
115                     currentBuffer.length << 1,
116                     newcount - filledBufferSum);
117                 filledBufferSum += currentBuffer.length;
118             }
119 
120             currentBufferIndex++;
121             currentBuffer = new byte[newBufferSize];
122             buffers.add(currentBuffer);
123         }
124     }

126     /**
127      * Write the bytes to byte array.
128      * @param b the bytes to write
129      * @param off The start offset
130      * @param len The number of bytes to write
131      */
132     @Override
133     public void write(final byte[] b, final int off, final int len) {
134         if ((off < 0)
135                 || (off > b.length)
136                 || (len < 0)
137                 || ((off + len) > b.length)
138                 || ((off + len) < 0)) {
139             throw new IndexOutOfBoundsException();
140         } else if (len == 0) {
141             return;
142         }
143         synchronized (this) {
144             final int newcount = count + len;
145             int remaining = len;
146             int inBufferPos = count - filledBufferSum;
147             while (remaining > 0) {
148                 final int part = Math.min(remaining, currentBuffer.length - inBufferPos);
149                 System.arraycopy(b, off + len - remaining, currentBuffer, inBufferPos, part);
150                 remaining -= part;
151                 if (remaining > 0) {
152                     needNewBuffer(newcount);
153                     inBufferPos = 0;
154                 }
155             }
156             count = newcount;
157         }
158     }

160     /**
161      * Write a byte to byte array.
162      * @param b the byte to write
163      */
164     @Override
165     public synchronized void write(final int b) {
166         int inBufferPos = count - filledBufferSum;
167         if (inBufferPos == currentBuffer.length) {
168             needNewBuffer(count + 1);
169             inBufferPos = 0;
170         }
171         currentBuffer[inBufferPos] = (byte) b;
172         count++;
173     }

175     /**
176      * Writes the entire contents of the specified input stream to this
177      * byte stream. Bytes from the input stream are read directly into the
178      * internal buffers of this streams.
179      *
180      * @param in the input stream to read from
181      * @return total number of bytes read from the input stream
182      *         (and written to this stream)
183      * @throws java.io.IOException if an I/O error occurs while reading the input stream
184      * @since 1.4
185      */
186     public synchronized int write(final InputStream in) throws IOException {
187         int readCount = 0;
188         int inBufferPos = count - filledBufferSum;
189         int n = in.read(currentBuffer, inBufferPos, currentBuffer.length - inBufferPos);
190         while (n != -1) {
191             readCount += n;
192             inBufferPos += n;
193             count += n;
194             if (inBufferPos == currentBuffer.length) {
195                 needNewBuffer(currentBuffer.length);
196                 inBufferPos = 0;
197             }
198             n = in.read(currentBuffer, inBufferPos, currentBuffer.length - inBufferPos);
199         }
200         return readCount;
201     }

224     /**
225      * @see java.io.ByteArrayOutputStream#reset()
226      */
227     public synchronized void reset() {
228         count = 0;
229         filledBufferSum = 0;
230         currentBufferIndex = 0;
231         if (reuseBuffers) {
232             currentBuffer = buffers.get(currentBufferIndex);
233         } else {
234             //Throw away old buffers
235             currentBuffer = null;
236             int size = buffers.get(0).length;
237             buffers.clear();
238             needNewBuffer(size);
239             reuseBuffers = true;
240         }
241     }

243     /**
244      * Writes the entire contents of this byte stream to the
245      * specified output stream.
246      *
247      * @param out  the output stream to write to
248      * @throws java.io.IOException if an I/O error occurs, such as if the stream is closed
249      * @see java.io.ByteArrayOutputStream#writeTo(java.io.OutputStream)
250      */
251     public synchronized void writeTo(final OutputStream out) throws IOException {
252         int remaining = count;
253         for (final byte[] buf : buffers) {
254             final int c = Math.min(buf.length, remaining);
255             out.write(buf, 0, c);
256             remaining -= c;
257             if (remaining == 0) {
258                 break;
259             }
260         }
261     }

289     /**
290      * Fetches entire contents of an <code>InputStream</code> and represent
291      * same data as result InputStream.
292      * <p>
293      * This method is useful where,
294      * <ul>
295      * <li>Source InputStream is slow.</li>
296      * <li>It has network resources associated, so we cannot keep it open for
297      * long time.</li>
298      * <li>It has network timeout associated.</li>
299      * </ul>
300      * It can be used in favor of {@link #toByteArray()}, since it
301      * avoids unnecessary allocation and copy of byte[].<br>
302      * This method buffers the input internally, so there is no need to use a
303      * <code>BufferedInputStream</code>.
304      *
305      * @param input Stream to be fully buffered.
306      * @param size the initial buffer size
307      * @return A fully buffered stream.
308      * @throws java.io.IOException if an I/O error occurs
309      * @since 2.5
310      */
311     public static InputStream toBufferedInputStream(final InputStream input, int size)
312             throws IOException {
313         // It does not matter if a ByteArrayOutputStream is not closed as close() is a no-op
314         @SuppressWarnings("resource")
315         final ByteArrayOutputStream output = new ByteArrayOutputStream(size);
316         output.write(input);
317         return output.toInputStream();
318     }

320     /**
321      * Gets the current contents of this byte stream as a Input Stream. The
322      * returned stream is backed by buffers of <code>this</code> stream,
323      * avoiding memory allocation and copy, thus saving space and time.<br>
324      *
325      * @return the current contents of this output stream.
326      * @see java.io.ByteArrayOutputStream#toByteArray()
327      * @see #reset()
328      * @since 2.5
329      */
330     public synchronized InputStream toInputStream() {
331         int remaining = count;
332         if (remaining == 0) {
333             return new ClosedInputStream();
334         }
335         final List<ByteArrayInputStream> list = new ArrayList<ByteArrayInputStream>(buffers.size());
336         for (final byte[] buf : buffers) {
337             final int c = Math.min(buf.length, remaining);
338             list.add(new ByteArrayInputStream(buf, 0, c));
339             remaining -= c;
340             if (remaining == 0) {
341                 break;
342             }
343         }
344         reuseBuffers = false;
345         return new SequenceInputStream(Collections.enumeration(list));
346     }

348     /**
349      * Gets the curent contents of this byte stream as a byte array.
350      * The result is independent of this stream.
351      *
352      * @return the current contents of this output stream, as a byte array
353      * @see java.io.ByteArrayOutputStream#toByteArray()
354      */
355     public synchronized byte[] toByteArray() {
356         int remaining = count;
357         if (remaining == 0) {
358             return EMPTY_BYTE_ARRAY;
359         }
360         final byte newbuf[] = new byte[remaining];
361         int pos = 0;
362         for (final byte[] buf : buffers) {
363             final int c = Math.min(buf.length, remaining);
364             System.arraycopy(buf, 0, newbuf, pos, c);
365             pos += c;
366             remaining -= c;
367             if (remaining == 0) {
368                 break;
369             }
370         }
371         return newbuf;
372     }
