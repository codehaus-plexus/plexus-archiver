1   package org.codehaus.plexus.archiver.bzip2;

3   import static org.codehaus.plexus.archiver.util.Streams.fileInputStream;
4   
5   import java.io.File;
6   import java.io.IOException;
7   import java.io.InputStream;
8   import java.util.HashMap;
9   import javax.annotation.Nonnull;
10  import javax.annotation.WillNotClose;
11  
12  import org.codehaus.plexus.components.io.attributes.FileAttributes;
13  import org.codehaus.plexus.components.io.attributes.PlexusIoResourceAttributes;
14  import org.codehaus.plexus.components.io.resources.PlexusIoCompressedFileResourceCollection;
15  import org.codehaus.plexus.components.io.resources.PlexusIoResourceCollection;
16  
17  /**
18   * Implementation of {@link PlexusIoResourceCollection} for
19   * bzip2 compressed files.
20   */
21  public class PlexusIoBzip2ResourceCollection
22      extends PlexusIoCompressedFileResourceCollection
23  {
24  
25      @Nonnull
26      @Override
27      protected @WillNotClose
28      InputStream getInputStream( File file )
29          throws IOException
30      {
31          return BZip2UnArchiver.getBZip2InputStream( fileInputStream( file ) );
32      }
33  
34      @Override protected PlexusIoResourceAttributes getAttributes( File file ) throws IOException
35      {
36          return new FileAttributes( file, new HashMap<Integer, String>(), new HashMap<Integer, String>() );
37      }
38  
39      @Override
40      protected String getDefaultExtension()
41      {
42          return ".bz2";
43      }
44  
45  }
