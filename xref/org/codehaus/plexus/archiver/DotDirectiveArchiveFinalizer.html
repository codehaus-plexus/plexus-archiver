3   import java.io.BufferedReader;
4   import java.io.File;
5   import java.io.IOException;
6   import java.nio.charset.StandardCharsets;
7   import java.nio.file.Files;
8   import java.util.Collections;
9   import java.util.List;
10  
11  import org.codehaus.plexus.util.FileUtils;
12  import org.codehaus.plexus.util.StringUtils;
13  
14  /**
15   * An {@link ArchiveFinalizer} that process dot files with archiver directives
16   * contained within. This basically means you can communicate archive creation
17   * instructions between processes using dot files.
18   *
19   * @author Jason van Zyl
20   */
21  public class DotDirectiveArchiveFinalizer
22      extends AbstractArchiveFinalizer
23  {
24  
25      private static final String DEFAULT_DOT_FILE_PREFIX = ".plxarc";
26  
27      private final File dotFileDirectory;
28  
29      private final String dotFilePrefix;
30  
31      public DotDirectiveArchiveFinalizer( File dotFileDirectory )
32      {
33          this( dotFileDirectory, DEFAULT_DOT_FILE_PREFIX );
34      }
35  
36      public DotDirectiveArchiveFinalizer( File dotFileDirectory, String dotFilePrefix )
37      {
38          this.dotFileDirectory = dotFileDirectory;
39  
40          this.dotFilePrefix = dotFilePrefix;
41      }
42  
43      @Override
44      public void finalizeArchiveCreation( Archiver archiver )
45          throws ArchiverException
46      {
47          try
48          {
49              List<File> dotFiles = FileUtils.getFiles( dotFileDirectory, dotFilePrefix + "*", null );
50  
51              for ( File dotFile : dotFiles )
52              {
53                  try ( BufferedReader in = Files.newBufferedReader( dotFile.toPath(), StandardCharsets.UTF_8 ) )
54                  {
55  
56                      for ( String line = in.readLine(); line != null; line = in.readLine() )
57                      {
58                          String[] s = StringUtils.split( line, ":" );
59  
60                          if ( s.length == 1 )
61                          {
62                              File directory = new File( dotFileDirectory, s[0] );
63  
64                              System.out.println( "adding directory = " + directory );
65  
66                              archiver.addDirectory( directory );
67                          }
68                          else
69                          {
70                              File directory = new File( dotFileDirectory, s[0] );
71  
72                              System.out.println( "adding directory = " + directory + " to: " + s[1] );
73  
74                              if ( s[1].endsWith( "/" ) )
75                              {
76                                  archiver.addDirectory( directory, s[1] );
77                              }
78                              else
79                              {
80                                  archiver.addDirectory( directory, s[1] + "/" );
81                              }
82                          }
83                      }
84                  }
85              }
86          }
87          catch ( IOException e )
88          {
89              throw new ArchiverException( "Error processing dot files.", e );
90          }
91      }
92  
93      @Override
94      public List getVirtualFiles()
95      {
96          return Collections.EMPTY_LIST;
97      }
98  
99  }
