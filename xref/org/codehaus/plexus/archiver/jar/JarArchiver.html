290     private Manifest createManifest()
291         throws ArchiverException
292     {
293             Manifest finalManifest = Manifest.getDefaultManifest();
294 
295             if ( ( manifest == null ) && ( manifestFile != null ) )
296             {
297                 // if we haven't got the manifest yet, attempt to
298                 // get it now and have manifest be the final merge
299                 manifest = getManifest( manifestFile );
300             }
301 
302         /*
303         * Precedence: manifestFile wins over inline manifest,
304         * over manifests read from the filesets over the original
305         * manifest.
306         *
307         * merge with null argument is a no-op
308         */
309 
310         if ( isInUpdateMode() )
311         {
312             JdkManifestFactory.merge( finalManifest, originalManifest, false );
313         }
314         JdkManifestFactory.merge( finalManifest, filesetManifest, false );
315         JdkManifestFactory.merge( finalManifest, configuredManifest, false );
316         JdkManifestFactory.merge( finalManifest, manifest, !mergeManifestsMain );
317 
318         return finalManifest;
319     }

321     private void writeManifest( ConcurrentJarCreator zOut, Manifest manifest )
322         throws IOException, ArchiverException
323     {
324         for ( Enumeration e = manifest.getWarnings(); e.hasMoreElements(); )
325         {
326             getLogger().warn( "Manifest warning: " + e.nextElement() );
327         }
328 
329         zipDir( null, zOut, "META-INF/", DEFAULT_DIR_MODE, getEncoding());
330         // time to write the manifest
331         ByteArrayOutputStream baos = new ByteArrayOutputStream();
332         manifest.write( baos );
333 
334         ByteArrayInputStream bais = new ByteArrayInputStream( baos.toByteArray() );
335         super.zipFile( createInputStreamSupplier( bais ), zOut, MANIFEST_NAME, System.currentTimeMillis(), null,
336                        DEFAULT_FILE_MODE, null );
337         super.initZipOutputStream( zOut );
338     }

349     /**
350      * Create the index list to speed up classloading.
351      * This is a JDK 1.3+ specific feature and is enabled by default. See
352      * <a href="http://java.sun.com/j2se/1.3/docs/guide/jar/jar.html#JAR%20Index">
353      * the JAR index specification</a> for more details.
354      *
355      * @param zOut the zip stream representing the jar being built.
356      * @throws IOException thrown if there is an error while creating the
357      *                     index and adding it to the zip stream.
358      * @throws org.codehaus.plexus.archiver.ArchiverException
359      *                     .
360      */
361     private void createIndexList( ConcurrentJarCreator zOut )
362         throws IOException, ArchiverException
363     {
364         ByteArrayOutputStream baos = new ByteArrayOutputStream();
365         // encoding must be UTF8 as specified in the specs.
366         PrintWriter writer = new PrintWriter( new OutputStreamWriter( baos, "UTF8" ) );
367 
368         // version-info blankline
369         writer.println( "JarIndex-Version: 1.0" );
370         writer.println();
371 
372         // header newline
373         writer.println( getDestFile().getName() );
374 
375         // filter out META-INF if it doesn't contain anything other than the index and manifest.
376         // this is what sun.misc.JarIndex does, guess we ought to be consistent.
377         Set<String> filteredDirs = addedDirs.allAddedDirs();
378         // our added dirs always have a trailing slash
379         if ( filteredDirs.contains( META_INF_NAME + '/' ) )
380         {
381             boolean add = false;
382             for ( String entry : entries.keySet() )
383             {
384                 if ( entry.startsWith( META_INF_NAME + '/' ) && !entry.equals( INDEX_NAME ) && !entry.equals(
385                     MANIFEST_NAME ) )
386                 {
387                     add = true;
388                     break;
389                 }
390             }
391             if ( !add )
392             {
393                 filteredDirs.remove( META_INF_NAME + '/' );
394             }
395         }
396         writeIndexLikeList( new ArrayList<String>( filteredDirs ), rootEntries, writer );
397         writer.println();
398 
399         if ( indexJars != null )
400         {
401             java.util.jar.Manifest mf = createManifest();
402             String classpath = mf.getMainAttributes().getValue( ManifestConstants.ATTRIBUTE_CLASSPATH );
403             String[] cpEntries = null;
404             if ( classpath != null )
405             {
406                 StringTokenizer tok = new StringTokenizer( classpath, " " );
407                 cpEntries = new String[tok.countTokens()];
408                 int c = 0;
409                 while ( tok.hasMoreTokens() )
410                 {
411                     cpEntries[c++] = tok.nextToken();
412                 }
413             }
414 
415             for ( String indexJar : indexJars )
416             {
417                 String name = findJarName( indexJar, cpEntries );
418                 if ( name != null )
419                 {
420                     ArrayList<String> dirs = new ArrayList<String>();
421                     ArrayList<String> files = new ArrayList<String>();
422                     grabFilesAndDirs( indexJar, dirs, files );
423                     if ( dirs.size() + files.size() > 0 )
424                     {
425                         writer.println( name );
426                         writeIndexLikeList( dirs, files, writer );
427                         writer.println();
428                     }
429                 }
430             }
431         }
432 
433         writer.flush();
434 
435         ByteArrayInputStream bais = new ByteArrayInputStream( baos.toByteArray() );
436 
437         super.zipFile( createInputStreamSupplier( bais ), zOut, INDEX_NAME, System.currentTimeMillis(), null,
438                        DEFAULT_FILE_MODE, null );
439     }

441     /**
442      * Overridden from Zip class to deal with manifests and index lists.
443      */
444     protected void zipFile( InputStreamSupplier is, ConcurrentJarCreator zOut, String vPath,
445                             long lastModified, File fromArchive,
446                             int mode, String symlinkDestination )
447         throws IOException, ArchiverException
448     {
449         if ( MANIFEST_NAME.equalsIgnoreCase( vPath ) )
450         {
451             if ( !doubleFilePass || skipWriting )
452             {
453                 filesetManifest( fromArchive, is.get() );
454             }
455         }
456         else if ( INDEX_NAME.equalsIgnoreCase( vPath ) && index )
457         {
458             getLogger().warn( "Warning: selected " + archiveType + " files include a META-INF/INDEX.LIST which will"
459                                   + " be replaced by a newly generated one." );
460         }
461         else
462         {
463             if ( index && ( !vPath.contains( "/" ) ) )
464             {
465                 rootEntries.addElement( vPath );
466             }
467             super.zipFile( is, zOut, vPath, lastModified, fromArchive, mode, symlinkDestination );
468         }
469     }

471     private void filesetManifest( File file, InputStream is )
472         throws ArchiverException
473     {
474         if ( ( manifestFile != null ) && manifestFile.equals( file ) )
475         {
476             // If this is the same name specified in 'manifest', this
477             // is the manifest to use
478             getLogger().debug( "Found manifest " + file );
479             if ( is != null )
480             {
481                 manifest = getManifest( is );
482             }
483             else
484             {
485                 manifest = getManifest( file );
486             }
487         }
488         else if ( ( filesetManifestConfig != null ) && filesetManifestConfig != FilesetManifestConfig.skip )
489         {
490             // we add this to our group of fileset manifests
491             getLogger().debug( "Found manifest to merge in file " + file );
492 
493             Manifest newManifest;
494             if ( is != null )
495             {
496                 newManifest = getManifest( is );
497             }
498             else
499             {
500                 newManifest = getManifest( file );
501             }
502 
503             if ( filesetManifest == null )
504             {
505                 filesetManifest = newManifest;
506             }
507             else
508             {
509                 JdkManifestFactory.merge( filesetManifest, newManifest, false );
510             }
511         }
512     }

514     /**
515      */
516     protected boolean createEmptyZip( File zipFile )
517         throws ArchiverException
518     {
519         if ( !createEmpty )
520         {
521             return true;
522         }
523 
524         try
525         {
526             getLogger().debug( "Building MANIFEST-only jar: " + getDestFile().getAbsolutePath() );
527             zipArchiveOutputStream =
528                 new ZipArchiveOutputStream( bufferedOutputStream( fileOutputStream( getDestFile(), "jar" ) ) );
529 
530             zipArchiveOutputStream.setEncoding( getEncoding() );
531             if ( isCompress() )
532             {
533                 zipArchiveOutputStream.setMethod( ZipArchiveOutputStream.DEFLATED );
534             }
535             else
536             {
537                 zipArchiveOutputStream.setMethod( ZipArchiveOutputStream.STORED );
538             }
539 			ConcurrentJarCreator ps = new ConcurrentJarCreator(Runtime.getRuntime().availableProcessors());
540             initZipOutputStream( ps );
541             finalizeZipOutputStream( ps );
542         }
543         catch ( IOException ioe )
544         {
545             throw new ArchiverException( "Could not create almost empty JAR archive (" + ioe.getMessage() + ")", ioe );
546         }
547         finally
548         {
549             // Close the output stream.
550             //IOUtil.close( zOut );
551             createEmpty = false;
552         }
553         return true;
554     }

600     /**
601      * Writes the directory entries from the first and the filenames
602      * from the second list to the given writer, one entry per line.
603      *
604      * @param dirs   The directories
605      * @param files  The files
606      * @param writer The printwriter ;)
607      */
608     protected final void writeIndexLikeList( List<String> dirs, List<String> files, PrintWriter writer )
609     {
610         // JarIndex is sorting the directories by ascending order.
611         // it has no value but cosmetic since it will be read into a
612         // hashtable by the classloader, but we'll do so anyway.
613         Collections.sort( dirs );
614         Collections.sort( files );
615         Iterator iter = dirs.iterator();
616         while ( iter.hasNext() )
617         {
618             String dir = (String) iter.next();
619 
620             // try to be smart, not to be fooled by a weird directory name
621             dir = dir.replace( '\\', '/' );
622             if ( dir.startsWith( "./" ) )
623             {
624                 dir = dir.substring( 2 );
625             }
626             while ( dir.startsWith( "/" ) )
627             {
628                 dir = dir.substring( 1 );
629             }
630             int pos = dir.lastIndexOf( '/' );
631             if ( pos != -1 )
632             {
633                 dir = dir.substring( 0, pos );
634             }
635 
636             // name newline
637             writer.println( dir );
638         }
639 
640         iter = files.iterator();
641         while ( iter.hasNext() )
642         {
643             writer.println( iter.next() );
644         }
645     }

647     /**
648      * try to guess the name of the given file.
649      * <p/>
650      * <p>If this jar has a classpath attribute in its manifest, we
651      * can assume that it will only require an index of jars listed
652      * there.  try to find which classpath entry is most likely the
653      * one the given file name points to.</p>
654      * <p/>
655      * <p>In the absence of a classpath attribute, assume the other
656      * files will be placed inside the same directory as this jar and
657      * use their basename.</p>
658      * <p/>
659      * <p>if there is a classpath and the given file doesn't match any
660      * of its entries, return null.</p>
661      *
662      * @param fileName  .
663      * @param classpath .
664      * @return The guessed name
665      */
666     protected static String findJarName( String fileName, String[] classpath )
667     {
668         if ( classpath == null )
669         {
670             return new File( fileName ).getName();
671         }
672         fileName = fileName.replace( File.separatorChar, '/' );
673         SortedMap<String, String> matches = new TreeMap<String, String>( new Comparator<String>()
674         {
675             // longest match comes first
676             public int compare( String o1, String o2 )
677             {
678                 if ( ( o1 != null ) && ( o2 != null ) )
679                 {
680                     return o2.length() - o1.length();
681                 }
682                 return 0;
683             }
684         } );
685 
686         for ( String aClasspath : classpath )
687         {
688             if ( fileName.endsWith( aClasspath ) )
689             {
690                 matches.put( aClasspath, aClasspath );
691             }
692             else
693             {
694                 int slash = aClasspath.indexOf( "/" );
695                 String candidate = aClasspath;
696                 while ( slash > -1 )
697                 {
698                     candidate = candidate.substring( slash + 1 );
699                     if ( fileName.endsWith( candidate ) )
700                     {
701                         matches.put( candidate, aClasspath );
702                         break;
703                     }
704                     slash = candidate.indexOf( "/" );
705                 }
706             }
707         }
708 
709         return matches.size() == 0 ? null : matches.get( matches.firstKey() );
710     }

712     /**
713      * Grab lists of all root-level files and all directories
714      * contained in the given archive.
715      *
716      * @param file  .
717      * @param files .
718      * @param dirs  .
719      * @throws java.io.IOException .
720      */
721     protected static void grabFilesAndDirs( String file, List<String> dirs, List<String> files )
722         throws IOException
723     {
724         File zipFile = new File( file );
725         if ( !zipFile.exists() )
726         {
727             Logger logger = new ConsoleLogger( Logger.LEVEL_INFO, "console" );
728             logger.error( "JarArchive skipping non-existing file: " + zipFile.getAbsolutePath() );
729         }
730         else if ( zipFile.isDirectory() )
731         {
732             Logger logger = new ConsoleLogger( Logger.LEVEL_INFO, "console" );
733             logger.info( "JarArchiver skipping indexJar " + zipFile + " because it is not a jar" );
734         }
735         else
736         {
737             org.apache.commons.compress.archivers.zip.ZipFile zf = null;
738             try
739             {
740                 zf = new org.apache.commons.compress.archivers.zip.ZipFile( file, "utf-8" );
741                 Enumeration<ZipArchiveEntry> entries = zf.getEntries();
742                 HashSet<String> dirSet = new HashSet<String>();
743                 while ( entries.hasMoreElements() )
744                 {
745                     ZipArchiveEntry ze = entries.nextElement();
746                     String name = ze.getName();
747                     // avoid index for manifest-only jars.
748                     if ( !name.equals( META_INF_NAME ) && !name.equals( META_INF_NAME + '/' ) && !name.equals(
749                         INDEX_NAME ) && !name.equals( MANIFEST_NAME ) )
750                     {
751                         if ( ze.isDirectory() )
752                         {
753                             dirSet.add( name );
754                         }
755                         else if ( !name.contains( "/" ) )
756                         {
757                             files.add( name );
758                         }
759                         else
760                         {
761                             // a file, not in the root
762                             // since the jar may be one without directory
763                             // entries, add the parent dir of this file as
764                             // well.
765                             dirSet.add( name.substring( 0, name.lastIndexOf( "/" ) + 1 ) );
766                         }
767                     }
768                 }
