338     /**
339      * Creates the manifest to be added to the JAR archive.
340      * Sub-classes may choose to override this method
341      * in order to inspect or modify the JAR manifest file.
342      *
343      * @return the manifest for the JAR archive.
344      *
345      * @throws ArchiverException
346      */
347     protected Manifest createManifest()
348         throws ArchiverException
349     {
350         Manifest finalManifest = Manifest.getDefaultManifest( minimalDefaultManifest );
351 
352         if ( ( manifest == null ) && ( manifestFile != null ) )
353         {
354             // if we haven't got the manifest yet, attempt to
355             // get it now and have manifest be the final merge
356             manifest = getManifest( manifestFile );
357         }
358 
359         /*
360          * Precedence: manifestFile wins over inline manifest,
361          * over manifests read from the filesets over the original
362          * manifest.
363          *
364          * merge with null argument is a no-op
365          */
366         if ( isInUpdateMode() )
367         {
368             JdkManifestFactory.merge( finalManifest, originalManifest, false );
369         }
370         JdkManifestFactory.merge( finalManifest, filesetManifest, false );
371         JdkManifestFactory.merge( finalManifest, configuredManifest, false );
372         JdkManifestFactory.merge( finalManifest, manifest, !mergeManifestsMain );
373 
374         return finalManifest;
375     }

377     private void writeManifest( ConcurrentJarCreator zOut, Manifest manifest )
378         throws IOException, ArchiverException
379     {
380         for ( Enumeration<String> e = manifest.getWarnings(); e.hasMoreElements(); )
381         {
382             getLogger().warn( "Manifest warning: " + e.nextElement() );
383         }
384 
385         zipDir( null, zOut, "META-INF/", DEFAULT_DIR_MODE, getEncoding() );
386 
387         // time to write the manifest
388         ByteArrayOutputStream baos = new ByteArrayOutputStream( 128 );
389         manifest.write( baos );
390         InputStreamSupplier in = () -> new ByteArrayInputStream( baos.toByteArray() );
391 
392         super.zipFile( in, zOut, MANIFEST_NAME, System.currentTimeMillis(), null, DEFAULT_FILE_MODE, null,
393                        false );
394         super.initZipOutputStream( zOut );
395     }

407     /**
408      * Create the index list to speed up classloading.
409      * This is a JDK 1.3+ specific feature and is enabled by default. See
410      * <a href="http://java.sun.com/j2se/1.3/docs/guide/jar/jar.html#JAR%20Index">
411      * the JAR index specification</a> for more details.
412      *
413      * @param zOut the zip stream representing the jar being built.
414      *
415      * @throws IOException thrown if there is an error while creating the
416      * index and adding it to the zip stream.
417      * @throws org.codehaus.plexus.archiver.ArchiverException
418      * @deprecated See <a href="https://bugs.openjdk.org/browse/JDK-8302819">JDK-8302819</a>
419      */
420     @Deprecated
421     private void createIndexList( ConcurrentJarCreator zOut )
422         throws IOException, ArchiverException
423     {
424         ByteArrayOutputStream baos = new ByteArrayOutputStream( 128 );
425         // encoding must be UTF8 as specified in the specs.
426         PrintWriter writer = new PrintWriter( new OutputStreamWriter( baos, StandardCharsets.UTF_8 ) );
427 
428         // version-info blankline
429         writer.println( "JarIndex-Version: 1.0" );
430         writer.println();
431 
432         // header newline
433         writer.println( getDestFile().getName() );
434 
435         // filter out META-INF if it doesn't contain anything other than the index and manifest.
436         // this is what sun.misc.JarIndex does, guess we ought to be consistent.
437         Set<String> filteredDirs = addedDirs.allAddedDirs();
438         // our added dirs always have a trailing slash
439         if ( filteredDirs.contains( META_INF_NAME + '/' ) )
440         {
441             boolean add = false;
442             for ( String entry : entries.keySet() )
443             {
444                 if ( entry.startsWith( META_INF_NAME + '/' ) && !entry.equals( INDEX_NAME )
445                          && !entry.equals( MANIFEST_NAME ) )
446                 {
447                     add = true;
448                     break;
449                 }
450             }
451             if ( !add )
452             {
453                 filteredDirs.remove( META_INF_NAME + '/' );
454             }
455         }
456         writeIndexLikeList( new ArrayList<>( filteredDirs ), rootEntries, writer );
457         writer.println();
458 
459         if ( indexJars != null )
460         {
461             java.util.jar.Manifest mf = createManifest();
462             String classpath = mf.getMainAttributes().getValue( ManifestConstants.ATTRIBUTE_CLASSPATH );
463             String[] cpEntries = null;
464             if ( classpath != null )
465             {
466                 StringTokenizer tok = new StringTokenizer( classpath, " " );
467                 cpEntries = new String[ tok.countTokens() ];
468                 int c = 0;
469                 while ( tok.hasMoreTokens() )
470                 {
471                     cpEntries[c++] = tok.nextToken();
472                 }
473             }
474 
475             for ( String indexJar : indexJars )
476             {
477                 String name = findJarName( indexJar, cpEntries );
478                 if ( name != null )
479                 {
480                     List<String> dirs = new ArrayList<>();
481                     List<String> files = new ArrayList<>();
482                     grabFilesAndDirs( indexJar, dirs, files );
483                     if ( dirs.size() + files.size() > 0 )
484                     {
485                         writer.println( name );
486                         writeIndexLikeList( dirs, files, writer );
487                         writer.println();
488                     }
489                 }
490             }
491         }
492 
493         writer.flush();
494 
495         InputStreamSupplier in = () -> new ByteArrayInputStream( baos.toByteArray() );
496 
497         super.zipFile( in, zOut, INDEX_NAME, System.currentTimeMillis(), null,
498                        DEFAULT_FILE_MODE, null, true );
499     }

501     /**
502      * Overridden from Zip class to deal with manifests and index lists.
503      */
504     @Override
505     protected void zipFile( InputStreamSupplier is, ConcurrentJarCreator zOut, String vPath,
506                             long lastModified, File fromArchive,
507                             int mode, String symlinkDestination, boolean addInParallel )
508         throws IOException, ArchiverException
509     {
510         if ( MANIFEST_NAME.equalsIgnoreCase( vPath ) )
511         {
512             if ( !doubleFilePass || skipWriting )
513             {
514                 try ( InputStream manifestInputStream = is.get() )
515                 {
516                     filesetManifest( fromArchive, manifestInputStream );
517                 }
518             }
519         }
520         else if ( INDEX_NAME.equalsIgnoreCase( vPath ) && index )
521         {
522             getLogger().warn( "Warning: selected " + archiveType + " files include a META-INF/INDEX.LIST which will"
523                                   + " be replaced by a newly generated one." );
524         }
525         else
526         {
527             if ( index && !vPath.contains( "/" ) )
528             {
529                 rootEntries.add( vPath );
530             }
531             super.zipFile( is, zOut, vPath, lastModified, fromArchive, mode, symlinkDestination, addInParallel );
532         }
533     }

535     private void filesetManifest( File file, InputStream is )
536         throws ArchiverException
537     {
538         if ( ( manifestFile != null ) && manifestFile.equals( file ) )
539         {
540             // If this is the same name specified in 'manifest', this
541             // is the manifest to use
542             getLogger().debug( "Found manifest " + file );
543             if ( is != null )
544             {
545                 manifest = getManifest( is );
546             }
547             else
548             {
549                 manifest = getManifest( file );
550             }
551         }
552         else if ( ( filesetManifestConfig != null ) && filesetManifestConfig != FilesetManifestConfig.skip )
553         {
554             // we add this to our group of fileset manifests
555             getLogger().debug( "Found manifest to merge in file " + file );
556 
557             Manifest newManifest;
558             if ( is != null )
559             {
560                 newManifest = getManifest( is );
561             }
562             else
563             {
564                 newManifest = getManifest( file );
565             }
566 
567             if ( filesetManifest == null )
568             {
569                 filesetManifest = newManifest;
570             }
571             else
572             {
573                 JdkManifestFactory.merge( filesetManifest, newManifest, false );
574             }
575         }
576     }

578     @Override
579     protected boolean createEmptyZip( File zipFile )
580         throws ArchiverException
581     {
582         if ( !createEmpty )
583         {
584             return true;
585         }
586 
587         try
588         {
589             getLogger().debug( "Building MANIFEST-only jar: " + getDestFile().getAbsolutePath() );
590             zipArchiveOutputStream =
591                 new ZipArchiveOutputStream( bufferedOutputStream( fileOutputStream( getDestFile(), "jar" ) ) );
592 
593             zipArchiveOutputStream.setEncoding( getEncoding() );
594             if ( isCompress() )
595             {
596                 zipArchiveOutputStream.setMethod( ZipArchiveOutputStream.DEFLATED );
597             }
598             else
599             {
600                 zipArchiveOutputStream.setMethod( ZipArchiveOutputStream.STORED );
601             }
602             ConcurrentJarCreator ps =
603                 new ConcurrentJarCreator( isRecompressAddedZips(), Runtime.getRuntime().availableProcessors() );
604             initZipOutputStream( ps );
605             finalizeZipOutputStream( ps );
606         }
607         catch ( IOException ioe )
608         {
609             throw new ArchiverException( "Could not create almost empty JAR archive (" + ioe.getMessage() + ")", ioe );
610         }
611         finally
612         {
613             // Close the output stream.
614             //IOUtil.close( zOut );
615             createEmpty = false;
616         }
617         return true;
618     }

668     /**
669      * Writes the directory entries from the first and the filenames
670      * from the second list to the given writer, one entry per line.
671      *
672      * @param dirs The directories
673      * @param files The files
674      * @param writer The printwriter ;)
675      * @deprecated See <a href="https://bugs.openjdk.org/browse/JDK-8302819">JDK-8302819</a>
676      */
677     @Deprecated
678     protected final void writeIndexLikeList( List<String> dirs, List<String> files, PrintWriter writer )
679     {
680         // JarIndex is sorting the directories by ascending order.
681         // it has no value but cosmetic since it will be read into a
682         // hashtable by the classloader, but we'll do so anyway.
683         Collections.sort( dirs );
684         Collections.sort( files );
685         for ( String dir : dirs )
686         {
687             // try to be smart, not to be fooled by a weird directory name
688             dir = dir.replace( '\\', '/' );
689             if ( dir.startsWith( "./" ) )
690             {
691                 dir = dir.substring( 2 );
692             }
693             while ( dir.startsWith( "/" ) )
694             {
695                 dir = dir.substring( 1 );
696             }
697             int pos = dir.lastIndexOf( '/' );
698             if ( pos != -1 )
699             {
700                 dir = dir.substring( 0, pos );
701             }
702 
703             // name newline
704             writer.println( dir );
705         }
706 
707         for ( String file : files )
708         {
709             writer.println( file );
710         }
711     }

713     /**
714      * try to guess the name of the given file.
715      * <p>
716      * If this jar has a classpath attribute in its manifest, we
717      * can assume that it will only require an index of jars listed
718      * there. try to find which classpath entry is most likely the
719      * one the given file name points to.</p>
720      * <p>
721      * In the absence of a classpath attribute, assume the other
722      * files will be placed inside the same directory as this jar and
723      * use their basename.</p>
724      * <p>
725      * if there is a classpath and the given file doesn't match any
726      * of its entries, return null.</p>
727      *
728      * @param fileName .
729      * @param classpath .
730      *
731      * @return The guessed name
732      */
733     protected static String findJarName( String fileName, String[] classpath )
734     {
735         if ( classpath == null )
736         {
737             return new File( fileName ).getName();
738         }
739         fileName = fileName.replace( File.separatorChar, '/' );
740 
741         // longest match comes first
742         SortedMap<String, String> matches = new TreeMap<>( Comparator.comparingInt( String::length ).reversed() );
743 
744         for ( String aClasspath : classpath )
745         {
746             if ( fileName.endsWith( aClasspath ) )
747             {
748                 matches.put( aClasspath, aClasspath );
749             }
750             else
751             {
752                 int slash = aClasspath.indexOf( "/" );
753                 String candidate = aClasspath;
754                 while ( slash > -1 )
755                 {
756                     candidate = candidate.substring( slash + 1 );
757                     if ( fileName.endsWith( candidate ) )
758                     {
759                         matches.put( candidate, aClasspath );
760                         break;
761                     }
762                     slash = candidate.indexOf( "/" );
763                 }
764             }
765         }
766 
767         return matches.size() == 0 ? null : matches.get( matches.firstKey() );
768     }

770     /**
771      * Grab lists of all root-level files and all directories
772      * contained in the given archive.
773      *
774      * @param file .
775      * @param files .
776      * @param dirs .
777      *
778      * @throws java.io.IOException
779      */
780     private void grabFilesAndDirs( String file, List<String> dirs, List<String> files )
781         throws IOException
782     {
783         File zipFile = new File( file );
784         if ( !zipFile.exists() )
785         {
786             getLogger().error( "JarArchive skipping non-existing file: " + zipFile.getAbsolutePath() );
787         }
788         else if ( zipFile.isDirectory() )
789         {
790             getLogger().info( "JarArchiver skipping indexJar " + zipFile + " because it is not a jar" );
791         }
792         else
793         {
794             try ( ZipFile zf = new ZipFile( file, "utf-8" ) )
795             {
796                 Enumeration<ZipArchiveEntry> entries = zf.getEntries();
797                 HashSet<String> dirSet = new HashSet<>();
798                 while ( entries.hasMoreElements() )
799                 {
800                     ZipArchiveEntry ze = entries.nextElement();
801                     String name = ze.getName();
802                     // avoid index for manifest-only jars.
803                     if ( !name.equals( META_INF_NAME ) && !name.equals( META_INF_NAME + '/' ) && !name.equals(
804                         INDEX_NAME ) && !name.equals( MANIFEST_NAME ) )
805                     {
806                         if ( ze.isDirectory() )
807                         {
808                             dirSet.add( name );
809                         }
810                         else if ( !name.contains( "/" ) )
811                         {
812                             files.add( name );
813                         }
814                         else
815                         {
816                             // a file, not in the root
817                             // since the jar may be one without directory
818                             // entries, add the parent dir of this file as
819                             // well.
820                             dirSet.add( name.substring( 0, name.lastIndexOf( "/" ) + 1 ) );
821                         }
822                     }
823                 }
824                 dirs.addAll( dirSet );
825             }
826         }
827     }

829     /**
830      * Override the behavior of the Zip Archiver to match the output of the JAR tool.
831      *
832      * @param zipEntry to set the last modified time
833      * @param lastModifiedTime to set in the zip entry only if {@link #getLastModifiedTime()} returns null
834      */
835     @Override
836     protected void setZipEntryTime( ZipArchiveEntry zipEntry, long lastModifiedTime )
837     {
838         if ( getLastModifiedTime() != null )
839         {
840             lastModifiedTime = getLastModifiedTime().toMillis();
841         }
842 
843         // The JAR tool does not round up, so we keep that behavior here (JDK-8277755).
844         zipEntry.setTime( lastModifiedTime );
845     }
